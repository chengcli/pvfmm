20template <class Real_t>
21void Cheb_Node<Real_t>::Initialize(TreeNode* parent_, int path2node_, TreeNode::NodeData* data_) {
22 MPI_Node<Real_t>::Initialize(parent_,path2node_,data_);
23
24 //Set Cheb_Node specific data.
25 NodeData* cheb_data=dynamic_cast<NodeData*>(data_);
26 Cheb_Node<Real_t>* parent=dynamic_cast<Cheb_Node<Real_t>*>(this->Parent());
27 if(cheb_data!=NULL){
28 cheb_deg=cheb_data->cheb_deg;
29 input_fn=cheb_data->input_fn;
30 data_dof=cheb_data->data_dof;
31 tol=cheb_data->tol;
32 }else if(parent!=NULL){
33 cheb_deg=parent->cheb_deg;
34 input_fn=parent->input_fn;
35 data_dof=parent->data_dof;
36 tol=parent->tol;
37 }
38
39 //Compute Chebyshev approximation.
40 if(this->IsLeaf() && !this->IsGhost()){
41 if(!input_fn.IsEmpty() && data_dof>0){
42 Real_t s=sctl::pow<Real_t>(0.5,this->Depth());
43 int n1=sctl::pow<unsigned int>(cheb_deg+1,this->Dim());
44 std::vector<Real_t> coord=cheb_nodes<Real_t>(cheb_deg,this->Dim());
45 for(int i=0;i<n1;i++){
46 coord[i*3+0]=coord[i*3+0]*s+this->Coord()[0];
47 coord[i*3+1]=coord[i*3+1]*s+this->Coord()[1];
48 coord[i*3+2]=coord[i*3+2]*s+this->Coord()[2];
49 }
50
51 std::vector<Real_t> input_val(n1*data_dof);
52 input_fn(&coord[0],n1,&input_val[0]);
53 Matrix<Real_t> M_val(n1,data_dof,&input_val[0],false);
54 M_val=M_val.Transpose();
55
56 cheb_coeff.Resize(((cheb_deg+1)*(cheb_deg+2)*(cheb_deg+3))/6*data_dof); cheb_coeff.SetZero();
57 cheb_approx<Real_t,Real_t>(&input_val[0], cheb_deg, data_dof, &cheb_coeff[0]);
58 }else if(this->cheb_value.Dim()>0){
59 size_t n_ptr=this->cheb_coord.Dim()/this->Dim();
60 assert(n_ptr*data_dof==this->cheb_value.Dim());
61 points2cheb<Real_t>(cheb_deg,&(this->cheb_coord[0]),&(this->cheb_value[0]),
62 this->cheb_coord.Dim()/this->Dim(),data_dof,this->Coord(),
63 1/(Real_t)(1UL<<this->Depth()), cheb_coeff);
64 }
65 }

84template <class Real_t>
85bool Cheb_Node<Real_t>::SubdivCond(){
86 // Do not subdivide beyond max_depth
87 if(this->Depth()>=this->max_depth-1) return false;
88 if(!this->IsLeaf()){ // If has non-leaf children, then return true.
89 int n=(1UL<<this->Dim());
90 for(int i=0;i<n;i++){
91 Cheb_Node<Real_t>* ch=static_cast<Cheb_Node<Real_t>*>(this->Child(i));
92 assert(ch!=NULL); //This should never happen
93 if(!ch->IsLeaf() || ch->IsGhost()) return true;
94 }
95 }
96 else{ // Do not refine ghost leaf nodes.
97 if(this->IsGhost()) return false;
98 }
99 if(MPI_Node<Real_t>::SubdivCond()) return true;
100
101 if(!this->IsLeaf()){
102 std::vector<Real_t> val((cheb_deg+1)*(cheb_deg+1)*(cheb_deg+1)*data_dof);
103 std::vector<Real_t> x=cheb_nodes<Real_t>(cheb_deg,1);
104 std::vector<Real_t> y=x;
105 std::vector<Real_t> z=x;
106 Real_t s=sctl::pow<Real_t>(0.5,this->Depth());
107 Real_t* coord=this->Coord();
108 for(size_t i=0;i<x.size();i++){
109 x[i]=x[i]*s+coord[0];
110 y[i]=y[i]*s+coord[1];
111 z[i]=z[i]*s+coord[2];
112 }
113 read_val(x,y,z, cheb_deg+1, cheb_deg+1, cheb_deg+1, &val[0]);
114
115 std::vector<Real_t> tmp_coeff(data_dof*((cheb_deg+1)*(cheb_deg+2)*(cheb_deg+3))/6);
116 cheb_approx<Real_t,Real_t>(&val[0],cheb_deg,data_dof,&tmp_coeff[0]);
117 return (cheb_err(&(tmp_coeff[0]),cheb_deg,data_dof)>tol);
118 }else{
119 assert(cheb_coeff.Dim()>0);
120 Real_t err=cheb_err(&(cheb_coeff[0]),cheb_deg,data_dof);
121 return (err>tol);
122 }
123}

125template <class Real_t>
126void Cheb_Node<Real_t>::Subdivide() {
127 if(!this->IsLeaf()) return;
128 MPI_Node<Real_t>::Subdivide();
129 if(cheb_deg<0 || cheb_coeff.Dim()==0 || !input_fn.IsEmpty()) return;
130
131 std::vector<Real_t> x(cheb_deg+1);
132 std::vector<Real_t> y(cheb_deg+1);
133 std::vector<Real_t> z(cheb_deg+1);
134 Vector<Real_t> cheb_node=cheb_nodes<Real_t>(cheb_deg,1);
135 Vector<Real_t> val(sctl::pow<unsigned int>(cheb_deg+1,this->Dim())*data_dof);
136 Vector<Real_t> child_cheb_coeff[8];
137 int n=(1UL<<this->Dim());
138 for(int i=0;i<n;i++){
139 Real_t coord[3]={(Real_t)((i )%2?0:-1.0),
140 (Real_t)((i/2)%2?0:-1.0),
141 (Real_t)((i/4)%2?0:-1.0)};
142 for(int j=0;j<=cheb_deg;j++){
143 x[j]=cheb_node[j]+coord[0];
144 y[j]=cheb_node[j]+coord[1];
145 z[j]=cheb_node[j]+coord[2];
146 }
147 cheb_eval(cheb_coeff, cheb_deg, x, y, z, val);
148 assert(val.Dim()==sctl::pow<unsigned int>(cheb_deg+1,this->Dim())*data_dof);
149 child_cheb_coeff[i].Resize(data_dof*((cheb_deg+1)*(cheb_deg+2)*(cheb_deg+3))/6);
150 cheb_approx<Real_t,Real_t>(&val[0],cheb_deg,data_dof,&(child_cheb_coeff[i][0]));
151
152 Cheb_Node<Real_t>* child=static_cast<Cheb_Node<Real_t>*>(this->Child(i));
153 child->cheb_coeff=child_cheb_coeff[i];
154 assert(child->cheb_deg==cheb_deg);
155 assert(child->tol==tol);
156 }

193template <class VTUData_t, class Node_t>
194void Cheb_Node<Real_t>::VTU_Data(VTUData_t& vtu_data, std::vector<Node_t*>& nodes, int lod){
195 typedef typename VTUData_t::VTKReal_t VTKReal_t;
196 //MPI_Node<Real_t>::VTU_Data(vtu_data, nodes, lod);
197
198 VTUData_t new_data;
199 { // Set new data
200 new_data.value.resize(1);
201 new_data.name.push_back("cheb_value");
202 std::vector<VTKReal_t>& coord=new_data.coord;
203 std::vector<VTKReal_t>& value=new_data.value[0];
204
205 std::vector<int32_t>& connect=new_data.connect;
206 std::vector<int32_t>& offset =new_data.offset;
207 std::vector<uint8_t>& types =new_data.types;
208
209 int gridpt_cnt;
210 std::vector<Real_t> grid_pts;
211 {
212 int cheb_deg=lod-1;
213 std::vector<Real_t> cheb_node_;
214 if(cheb_deg>=0) cheb_node_=cheb_nodes<Real_t>(cheb_deg,1);
215 gridpt_cnt=cheb_node_.size()+2;
216 grid_pts.resize(gridpt_cnt);
217 for(size_t i=0;i<cheb_node_.size();i++) grid_pts[i+1]=cheb_node_[i];
218 grid_pts[0]=0.0; grid_pts[gridpt_cnt-1]=1.0;
219 }
220
221 Vector<Real_t> gridval;
222 for(size_t nid=0;nid<nodes.size();nid++){
223 Node_t* n=nodes[nid];
224 if(n->IsGhost() || !n->IsLeaf()) continue;
225
226 size_t point_cnt=coord.size()/PVFMM_COORD_DIM;
227 Real_t* c=n->Coord();
228 Real_t s=sctl::pow<Real_t>(0.5,n->Depth());
229 for(int i0=0;i0<gridpt_cnt;i0++)
230 for(int i1=0;i1<gridpt_cnt;i1++)
231 for(int i2=0;i2<gridpt_cnt;i2++){
232 coord.push_back(c[0]+grid_pts[i2]*s);
233 coord.push_back(c[1]+grid_pts[i1]*s);
234 coord.push_back(c[2]+grid_pts[i0]*s);
235 for(int j=0;j<n->data_dof;j++) value.push_back(0.0);
236 }
237 for(int i0=0;i0<(gridpt_cnt-1);i0++)
238 for(int i1=0;i1<(gridpt_cnt-1);i1++)
239 for(int i2=0;i2<(gridpt_cnt-1);i2++){
240 for(int j0=0;j0<2;j0++)
241 for(int j1=0;j1<2;j1++)
242 for(int j2=0;j2<2;j2++)
243 connect.push_back(point_cnt + ((i0+j0)*gridpt_cnt + (i1+j1))*gridpt_cnt + (i2+j2)*1);
244 offset.push_back(connect.size());
245 types.push_back(11);
246 }
247
248 {// Set point values.
249 if(gridval.Dim()!=(size_t)n->data_dof*gridpt_cnt*gridpt_cnt*gridpt_cnt)
250 gridval.ReInit(n->data_dof*gridpt_cnt*gridpt_cnt*gridpt_cnt);
251 std::vector<Real_t> x(gridpt_cnt);
252 std::vector<Real_t> y(gridpt_cnt);
253 std::vector<Real_t> z(gridpt_cnt);
254 for(int i=0;i<gridpt_cnt;i++){
255 x[i]=c[0]+s*grid_pts[i];
256 y[i]=c[1]+s*grid_pts[i];
257 z[i]=c[2]+s*grid_pts[i];
258 }
259 n->ReadVal(x, y, z, &gridval[0]);
260 //Rearrrange data
261 //(x1,x2,x3,...,y1,y2,...z1,...) => (x1,y1,z1,x2,y2,z2,...)
262 Matrix<VTKReal_t> M(n->data_dof,gridpt_cnt*gridpt_cnt*gridpt_cnt,&value[point_cnt*n->data_dof],false);
263 for(size_t i=0;i<gridval.Dim();i++) M[0][i]=gridval[i];
264 M=M.Transpose();
265 }
266 }
267 }
268 AppendVTUData(vtu_data, new_data);
269}

272void Cheb_Node<Real_t>::Gradient(){
273 int dim=3;//this->Dim();
274 if(this->IsLeaf() && ChebData().Dim()>0){
275 Real_t scale=sctl::pow<Real_t>(2,this->depth);
276 for(size_t i=0;i<ChebData().Dim();i++)
277 ChebData()[i]*=scale;
278
279 Vector<Real_t> coeff(ChebData().Dim()*dim);
280 cheb_grad(ChebData(),cheb_deg,coeff);
281 ChebData().Swap(coeff);
282 }
283 data_dof*=3;

287void Cheb_Node<Real_t>::Divergence(){
288 int dim=3;//this->Dim();
289 if(this->IsLeaf() && ChebData().Dim()>0){
290 assert(data_dof%3==0);
291 int n3=((cheb_deg+1)*(cheb_deg+2)*(cheb_deg+3))/6;
292 Vector<Real_t> coeff(ChebData().Dim()/dim);
293 for(int i=0;i<data_dof;i=i+dim)
294 cheb_div(&(ChebData()[n3*i]),cheb_deg,&coeff[n3*(i/dim)]);
295 ChebData().Swap(coeff);
296
297 Real_t scale=sctl::pow<Real_t>(2,this->depth);
298 for(size_t i=0;i<ChebData().Dim();i++)
299 ChebData()[i]*=scale;
300 }
301 data_dof/=dim;
302}

321template <class Real_t>
322void Cheb_Node<Real_t>::read_val(std::vector<Real_t> x,std::vector<Real_t> y, std::vector<Real_t> z, int nx, int ny, int nz, Real_t* val, bool show_ghost/*=true*/){
323 if(cheb_deg<0) return;
324 Real_t s=(Real_t)0.5*sctl::pow<Real_t>(0.5,this->Depth());
325 Real_t s_inv=1/s;
326 if(this->IsLeaf()){
327 if(cheb_coeff.Dim()!=(size_t)((cheb_deg+1)*(cheb_deg+2)*(cheb_deg+3))/6*data_dof
328 || (this->IsGhost() && !show_ghost)) return;
329 Vector<Real_t> out;
330 std::vector<Real_t> x_=x;
331 std::vector<Real_t> y_=y;
332 std::vector<Real_t> z_=z;
333 for(size_t i=0;i<x.size();i++)
334 x_[i]=(x[i]-this->Coord()[0])*s_inv-1;
335 for(size_t i=0;i<y.size();i++)
336 y_[i]=(y[i]-this->Coord()[1])*s_inv-1;
337 for(size_t i=0;i<z.size();i++)
338 z_[i]=(z[i]-this->Coord()[2])*s_inv-1;
339 cheb_eval(cheb_coeff, cheb_deg, x_, y_, z_, out);
340
341 for(int l=0;l<data_dof;l++)
342 for(size_t i=0;i<x.size();i++)
343 for(size_t j=0;j<y.size();j++)
344 for(size_t k=0;k<z.size();k++){
345 val[i+(j+(k+l*nz)*ny)*nx]=out[i+(j+(k+l*z.size())*y.size())*x.size()];
346 }
347 return;
348 }
349 Real_t coord_[3]={this->Coord()[0]+s, this->Coord()[1]+s, this->Coord()[2]+s};
350 Real_t* indx[3]={&(std::lower_bound(x.begin(),x.end(),coord_[0])[0]),
351 &(std::lower_bound(y.begin(),y.end(),coord_[1])[0]),
352 &(std::lower_bound(z.begin(),z.end(),coord_[2])[0])};
353 std::vector<Real_t> x1[2]={std::vector<Real_t>(&(x.begin()[0]),indx[0]),std::vector<Real_t>(indx[0],&(x.end()[0]))};
354 std::vector<Real_t> y1[2]={std::vector<Real_t>(&(y.begin()[0]),indx[1]),std::vector<Real_t>(indx[1],&(y.end()[0]))};
355 std::vector<Real_t> z1[2]={std::vector<Real_t>(&(z.begin()[0]),indx[2]),std::vector<Real_t>(indx[2],&(z.end()[0]))};
356
357 for(int i=0;i<8;i++){
358 std::vector<Real_t>& x1_=x1[i%2];
359 std::vector<Real_t>& y1_=y1[(i/2)%2];
360 std::vector<Real_t>& z1_=z1[(i/4)%2];
361 if(x1_.size()>0 && y1_.size()>0 && z1_.size()>0){
362 static_cast<Cheb_Node<Real_t>*>(this->Child(i))->read_val(x1_,y1_,z1_,nx,ny,nz,&val[(i%2?indx[0]-&x[0]:0)+((i/2)%2?indx[1]-&y[0]:0)*nx+((i/4)%2?indx[2]-&z[0]:0)*nx*ny],show_ghost);
363 }

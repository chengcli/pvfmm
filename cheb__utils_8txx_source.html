24/**
25 * \brief Returns the values of all chebyshev polynomials up to degree d,
26 * evaluated at points in the input vector. Output format:
27 * { T0[in[0]], ..., T0[in[n-1]], T1[in[0]], ..., T(d-1)[in[n-1]] }
28 */
29template <class T>
30inline void cheb_poly(int d, const T* in, int n, T* out){
31 if(d==0){
32 for(int i=0;i<n;i++)
33 out[i]=(sctl::fabs<T>(in[i])<=1?1.0:0);
34 }else if(d==1){
35 for(int i=0;i<n;i++){
36 out[i]=(sctl::fabs<T>(in[i])<=1?1.0:0);
37 out[i+n]=(sctl::fabs<T>(in[i])<=1?in[i]:0);
38 }
39 }else{
40 for(int j=0;j<n;j++){
41 T x=(sctl::fabs<T>(in[j])<=1?in[j]:0);
42 T y0=(sctl::fabs<T>(in[j])<=1?1.0:0);
43 out[j]=y0;
44 out[j+n]=x;
45
46 T y1=x;
47 T* y2=&out[2*n+j];
48 for(int i=2;i<=d;i++){
49 *y2=2*x*y1-y0;
50 y0=y1;
51 y1=*y2;
52 y2=&y2[n];
53 }
54 }
55 }
56}

60 * order polynomial as an estimate of error.
61 */
62template <class T>
63T cheb_err(T* cheb_coeff, int deg, int dof){
64 T err=0;
65 int indx=0;
66
67 for(int l=0;l<dof;l++)
68 for(int i=0;i<=deg;i++)
69 for(int j=0;i+j<=deg;j++)
70 for(int k=0;i+j+k<=deg;k++){
71 if(i+j+k==deg) err+=sctl::fabs<T>(cheb_coeff[indx]);
72 indx++;
73 }
74 return err;

88 * \brief Computes Chebyshev approximation from function values at cheb node points.
89 */
90template <class T, class Y>
91T cheb_approx(const T* fn_v, int cheb_deg, int dof, T* out, mem::MemoryManager* mem_mgr){
92 int d=cheb_deg+1;
93
94 // Precompute
95 Matrix<Y>* Mp=NULL;
96 static std::vector<Matrix<Y> > precomp;
97 #pragma omp critical(PVFMM_CHEB_APPROX)
98 {
99 if(precomp.size()<=(size_t)d){
100 precomp .resize(d+1);
101 }
102 if(precomp [d].Dim(0)==0 && precomp [d].Dim(1)==0){
103 std::vector<Y> x(d);
104 for(int i=0;i<d;i++)
105 x[i]=-sctl::cos<Y>((i+(T)0.5)*sctl::const_pi<T>()/d);
106
107 std::vector<Y> p(d*d);
108 cheb_poly(cheb_deg,&x[0],d,&p[0]);
109 for(int i=d;i<d*d;i++)
110 p[i]=p[i]*2;
111 for(int i=0;i<d*d;i++)
112 p[i]=p[i]/d;
113 Matrix<Y> Mp1(d,d,&p[0],false);
114 Matrix<Y> Mp1_=Mp1.Transpose();
115 precomp[d]=Mp1_;
116 }
117 Mp=&precomp[d];
118 }
119
120 // Create work buffers
121 size_t buff_size=dof*d*d*d;
122 Y* buff=mem::aligned_new<Y>(2*buff_size,mem_mgr);
123 Y* buff1=buff+buff_size*0;
124 Y* buff2=buff+buff_size*1;
125
126 Vector<Y> fn_v_in;
127 if(SameType<T,Y>()()){ // Initialize fn_v_in
128 fn_v_in.ReInit(d*d*d*dof,(T*)fn_v,false);
129 }else{
130 fn_v_in.ReInit(d*d*d*dof,buff1,false);
131 for(size_t i=0;i<fn_v_in.Dim();i++) fn_v_in[i]=fn_v[i];
132 }
133
134 { // Apply Mp along x-dimension
135 Matrix<Y> Mi(dof*d*d,d,&fn_v_in[0],false);
136 Matrix<Y> Mo(dof*d*d,d,buff2,false);
137 Mo=Mi*(*Mp);
138
139 Matrix<Y> Mo_t(d,dof*d*d,buff1,false);
140 for(size_t i=0;i<Mo.Dim(0);i++)
141 for(size_t j=0;j<Mo.Dim(1);j++){
142 Mo_t[j][i]=Mo[i][j];
143 }
144 }
145 { // Apply Mp along y-dimension
146 Matrix<Y> Mi(d*dof*d,d,buff1,false);
147 Matrix<Y> Mo(d*dof*d,d,buff2,false);
148 Mo=Mi*(*Mp);
149
150 Matrix<Y> Mo_t(d,d*dof*d,buff1,false);
151 for(size_t i=0;i<Mo.Dim(0);i++)
152 for(size_t j=0;j<Mo.Dim(1);j++){
153 Mo_t[j][i]=Mo[i][j];
154 }
155 }
156 { // Apply Mp along z-dimension
157 Matrix<Y> Mi(d*d*dof,d,buff1,false);
158 Matrix<Y> Mo(d*d*dof,d,buff2,false);
159 Mo=Mi*(*Mp);
160
161 Matrix<Y> Mo_t(d,d*d*dof,buff1,false);
162 for(size_t i=0;i<Mo.Dim(0);i++)
163 for(size_t j=0;j<Mo.Dim(1);j++){
164 Mo_t[j][i]=Mo[i][j];
165 }
166 }
167
168 { // Rearrange and write to out
169 int indx=0;
170 for(int l=0;l<dof;l++){
171 for(int i=0;i<d;i++){
172 for(int j=0;i+j<d;j++){
173 Y* buff_ptr=&buff1[(j+i*d)*d*dof+l];
174 for(int k=0;i+j+k<d;k++){
175 out[indx]=buff_ptr[k*dof];
176 indx++;
177 }
178 }
179 }
180 }
181 }
182
183 // Free memory
184 mem::aligned_delete<Y>(buff,mem_mgr);
185
186 return cheb_err(out,cheb_deg,dof);
187}

190 * \brief Returns the values of all legendre polynomials up to degree d,
191 * evaluated at points in the input vector. Output format:
192 * { P0[in[0]], ..., P0[in[n-1]], P1[in[0]], ..., P(d-1)[in[n-1]] }
193 */
194template <class T>
195inline void legn_poly(int d, T* in, int n, T* out){
196 if(d==0){
197 for(int i=0;i<n;i++)
198 out[i]=(sctl::fabs<T>(in[i])<=1?1.0:0);
199 }else if(d==1){
200 for(int i=0;i<n;i++){
201 out[i]=(sctl::fabs<T>(in[i])<=1?1.0:0);
202 out[i+n]=(sctl::fabs<T>(in[i])<=1?in[i]:0);
203 }
204 }else{
205 for(int j=0;j<n;j++){
206 T x=(sctl::fabs<T>(in[j])<=1?in[j]:0);
207 T y0=(sctl::fabs<T>(in[j])<=1?1.0:0);
208 out[j]=y0;
209 out[j+n]=x;
210
211 T y1=x;
212 T* y2=&out[2*n+j];
213 for(int i=2;i<=d;i++){
214 *y2=( (2*i-1)*x*y1-(i-1)*y0 )/i;
215 y0=y1;
216 y1=*y2;
217 y2=&y2[n];
218 }
219 }
220 }
221}

223/**
224 * \brief Computes Legendre-Gauss-Lobatto nodes and weights.
225 */
226template <class T>
227void gll_quadrature(int deg, T* x_, T* w){//*
228 T eps=sctl::machine_eps<T>()*64;
229 int d=deg+1;
230 assert(d>1);
231 int N=deg;
232
233 Vector<T> x(d,x_,false);
234 for(int i=0;i<d;i++)
235 x[i]=-sctl::cos<T>((sctl::const_pi<T>()*i)/N);
236 Matrix<T> P(d,d); P.SetZero();
237
238 T err=1;
239 Vector<T> xold(d);
240 while(err>eps){
241 xold=x;
242 for(int i=0;i<d;i++){
243 P[i][0]=1;
244 P[i][1]=x[i];
245 }
246 for(int k=2;k<=N;k++)
247 for(int i=0;i<d;i++)
248 P[i][k]=( (2*k-1)*x[i]*P[i][k-1]-(k-1)*P[i][k-2] )/k;
249 err=0;
250 for(int i=0;i<d;i++){
251 T dx=-( x[i]*P[i][N]-P[i][N-1] )/( d*P[i][N] );
252 err=(err<sctl::fabs<T>(dx)?sctl::fabs<T>(dx):err);
253 x[i]=xold[i]+dx;
254 }
255 }
256 for(int i=0;i<d;i++)
257 w[i]=2.0/(N*d*P[i][N]*P[i][N]);
258}

260/**
261 * \brief Computes Chebyshev approximation from function values at GLL points.
262 */
263template <class T, class Y>
264T gll2cheb(T* fn_v, int deg, int dof, T* out){//*
265 //T eps=sctl::machine_eps<T>()*64;
266
267 int d=deg+1;
268 static std::vector<Matrix<Y> > precomp;
269 static std::vector<Matrix<Y> > precomp_;
270 Matrix<Y>* Mp ;
271 Matrix<Y>* Mp_;
272 #pragma omp critical(PVFMM_GLL_TO_CHEB)
273 {
274 if(precomp.size()<=(size_t)d){
275 precomp .resize(d+1);
276 precomp_.resize(d+1);
277
278 std::vector<Y> x(d); //Cheb nodes.
279 for(int i=0;i<d;i++)
280 x[i]=-sctl::cos<Y>((i+(T)0.5)*sctl::const_pi<Y>()/d);
281
282 Vector<T> w(d);
283 Vector<T> x_legn(d); // GLL nodes.
284 gll_quadrature(deg, &x_legn[0], &w[0]);
285
286 Matrix<T> P(d,d); //GLL node 2 GLL coeff.
287 legn_poly(deg,&x_legn[0],d,&P[0][0]);
288 for(int i=0;i<d;i++)
289 for(int j=0;j<d;j++)
290 P[i][j]*=w[j]*0.5*(i<deg?(2*i+1):(i));
291
292 Matrix<T> M_gll2cheb(d,d); //GLL coeff 2 cheb node.
293 legn_poly(deg,&x[0],d,&M_gll2cheb[0][0]);
294
295 Matrix<T> M_g2c; //GLL node to cheb node.
296 M_g2c=M_gll2cheb.Transpose()*P;
297
298 std::vector<Y> p(d*d);
299 cheb_poly(deg,&x[0],d,&p[0]);
300 for(int i=0;i<d*d;i++)
301 p[i]=p[i]*2.0/d;
302 Matrix<Y> Mp1(d,d,&p[0],false);
303 Mp1=Mp1*M_g2c;
304
305 Matrix<Y> Mp1_=Mp1.Transpose();
306 precomp [d]=Mp1 ;
307 precomp_[d]=Mp1_;
308 }
309 Mp =&precomp [d];
310 Mp_=&precomp_[d];
311 }
312
313 std::vector<Y> fn_v0(d*d*d*dof);
314 std::vector<Y> fn_v1(d*d*d);
315 std::vector<Y> fn_v2(d*d*d);
316 std::vector<Y> fn_v3(d*d*d);
317
318 for(size_t i=0;i<(size_t)(d*d*d*dof);i++)
319 fn_v0[i]=fn_v[i];
320
321 int indx=0;
322 for(int l=0;l<dof;l++){
323 {
324 Matrix<Y> M0(d*d,d,&fn_v0[d*d*d*l],false);
325 Matrix<Y> M1(d*d,d,&fn_v1[0],false);
326 M1=M0*(*Mp_);
327 }
328 {
329 Matrix<Y> M0(d,d*d,&fn_v1[0],false);
330 Matrix<Y> M1(d,d*d,&fn_v2[0],false);
331 M1=(*Mp)*M0;
332 }
333 for(int i=0;i<d;i++){
334 Matrix<Y> M0(d,d,&fn_v2[d*d*i],false);
335 Matrix<Y> M1(d,d,&fn_v3[d*d*i],false);
336 M1=(*Mp)*M0;
337 }
338
339 for(int i=0;i<d;i++)
340 for(int j=0;j<d;j++){
341 fn_v3[i*d+j*d*d]/=2.0;
342 fn_v3[i+j*d*d]/=2.0;
343 fn_v3[i+j*d]/=2.0;
344 }
345 Y sum=0;
346 for(int i=0;i<d;i++)
347 for(int j=0;i+j<d;j++)
348 for(int k=0;i+j+k<d;k++){
349 sum+=sctl::fabs<T>(fn_v3[k+(j+i*d)*d]);
350 }
351 for(int i=0;i<d;i++)
352 for(int j=0;i+j<d;j++)
353 for(int k=0;i+j+k<d;k++){
354 out[indx]=fn_v3[k+(j+i*d)*d];
355 //if(sctl::fabs<T>(out[indx])<eps*sum) out[indx]=0;
356 indx++;
357 }
358 }
359
360 return cheb_err(out,deg,dof);
361}

504inline void cheb_eval(Vector<T>& coeff_, int cheb_deg, std::vector<T>& coord, Vector<T>& out){

507 int d=cheb_deg+1;
508 int n=coord.size()/dim;

515 for(int l=0;l<dof;l++)
516 for(int i=0;i<d;i++)
517 for(int j=0;i+j<d;j++)
518 for(int k=0;i+j+k<d;k++){

521 }

525 coord_=coord_.Transpose();
526

528 Matrix<T> py(d,n);
529 Matrix<T> pz(d,n);
530 cheb_poly(cheb_deg,&(coord_[0][0]),n,&(px[0][0]));
531 cheb_poly(cheb_deg,&(coord_[1][0]),n,&(py[0][0]));
532 cheb_poly(cheb_deg,&(coord_[2][0]),n,&(pz[0][0]));
533
534 Matrix<T> M_coeff0(d*d*dof, d, &coeff[0], false);
535 Matrix<T> M0 = (M_coeff0 * px).Transpose(); // {n, dof*d*d}
536
537 py = py.Transpose();
538 pz = pz.Transpose();
539 out.Resize(n*dof);
540 for(int i=0; i<n; i++)
541 for(int j=0; j<dof; j++){
542 Matrix<T> M0_ (d, d, &(M0[i][ j*d*d]), false);
543 Matrix<T> py_ (d, 1, &(py[i][ 0]), false);
544 Matrix<T> pz_ (1, d, &(pz[i][ 0]), false);
545
546 Matrix<T> M_out(1, 1, &( out[i*dof+j]), false);
547 M_out += pz_ * M0_ * py_;
548 }

598 * \param[in] coord Coordinates of points (x,y,z interleaved).

600 * \param[in] node_size Length of the side of the octant.

603template <class T>

606 int deg_=((int)(sctl::pow<T>(n*6,1.0/3.0)+0.5))/2;
607 deg_=(deg_>deg?deg:deg_);
608 deg_=(deg_>0?deg_:1);
609 int deg3=((deg_+1)*(deg_+2)*(deg_+3))/6;
610 cheb_coeff.Resize(dim*((deg+1)*(deg+2)*(deg+3))/6);
611 cheb_coeff.SetZero();

614 std::vector<T> coord_(n*3);
615 for(int i=0;i<n;i++){

617 coord_[i*3+1]=(coord[i*3+1]-node_coord[1])*2/node_size-1;

624 for(int i=0;i<n;i++)

641 indx1++;
642 }else{
643 cheb_coeff[indx]=0;
644 }
645 indx++;
646 }

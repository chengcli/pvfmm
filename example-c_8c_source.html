7void BiotSavart(const double* src_X, const double* src_V, long Ns, const double* trg_X, double* trg_V, long Nt) { // Direct evaluation
8 double oofp = 1 / (4 * M_PI);
9 #pragma omp parallel for schedule(static)
10 for (long t = 0; t < Nt; t++) {
11 trg_V[t*3+0] = 0;
12 trg_V[t*3+1] = 0;
13 trg_V[t*3+2] = 0;
14 for (long s = 0; s < Ns; s++) {
15 double X[3];
16 X[0] = trg_X[t*3+0] - src_X[s*3+0];
17 X[1] = trg_X[t*3+1] - src_X[s*3+1];
18 X[2] = trg_X[t*3+2] - src_X[s*3+2];
19 double r2 = X[0]*X[0] + X[1]*X[1] + X[2]*X[2];
20 double rinv = (r2 > 0 ? 1/sqrt(r2) : 0);
21 double rinv3 = rinv * rinv * rinv;
22
23 trg_V[t*3+0] += (src_V[s*3+1]*X[2] - src_V[s*3+2]*X[1]) * rinv3 * oofp;
24 trg_V[t*3+1] += (src_V[s*3+2]*X[0] - src_V[s*3+0]*X[2]) * rinv3 * oofp;
25 trg_V[t*3+2] += (src_V[s*3+0]*X[1] - src_V[s*3+1]*X[0]) * rinv3 * oofp;
26 }
27 }
28}

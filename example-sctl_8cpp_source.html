5template <class Real> void test_particle_fmm(const Comm& comm) {

7
8 Stokes3D_FSxU kernel_m2l;
9 Stokes3D_FxU kernel_sl;
10 Stokes3D_DxU kernel_dl;
11 srand48(comm.Rank());
12
13 // Create target and source vectors.
14 const Long N = 50000/comm.Size();
15 Vector<Real> trg_coord(N*DIM);
16 Vector<Real> sl_coord(N*DIM);
17 Vector<Real> dl_coord(N*DIM);
18 Vector<Real> dl_norml(N*DIM);
19 for (auto& a : trg_coord) a = (Real)(drand48()-0.5);
20 for (auto& a : sl_coord) a = (Real)(drand48()-0.5);
21 for (auto& a : dl_coord) a = (Real)(drand48()-0.5);
22 for (auto& a : dl_norml) a = (Real)(drand48()-0.5);
23 Long n_sl = sl_coord.Dim()/DIM;
24 Long n_dl = dl_coord.Dim()/DIM;
25
26 // Set source charges.
27 Vector<Real> sl_den(n_sl*kernel_sl.SrcDim());
28 Vector<Real> dl_den(n_dl*kernel_dl.SrcDim());
29 for (auto& a : sl_den) a = (Real)(drand48() - 0.5);
30 for (auto& a : dl_den) a = (Real)(drand48() - 0.5);
31
32 ParticleFMM<Real,DIM> fmm(comm);
33 fmm.SetAccuracy(10);
34 fmm.SetKernels(kernel_m2l, kernel_m2l, kernel_sl);
35 fmm.AddTrg("Potential", kernel_m2l, kernel_sl);
36 fmm.AddSrc("SingleLayer", kernel_sl, kernel_sl);
37 fmm.AddSrc("DoubleLayer", kernel_dl, kernel_dl);
38 fmm.SetKernelS2T("SingleLayer", "Potential",kernel_sl);
39 fmm.SetKernelS2T("DoubleLayer", "Potential",kernel_dl);
40
41 fmm.SetTrgCoord("Potential", trg_coord);
42 fmm.SetSrcCoord("SingleLayer", sl_coord);
43 fmm.SetSrcCoord("DoubleLayer", dl_coord, dl_norml);
44
45 fmm.SetSrcDensity("SingleLayer", sl_den);
46 fmm.SetSrcDensity("DoubleLayer", dl_den);
47
48 Vector<Real> Ufmm, Uref;
49 fmm.Eval(Ufmm, "Potential"); // Warm-up run
50 Ufmm = 0;
51
52 Profile::Enable(true);
53 Profile::Tic("FMM-Eval", &comm);
54 fmm.Eval(Ufmm, "Potential");
55 Profile::Toc();
56 Profile::Tic("Direct", &comm);

58 Profile::Toc();

61 Vector<Real> Uerr = Uref - Ufmm;
62 { // Print error
63 StaticArray<Real,2> loc_err{0,0}, glb_err{0,0};
64 for (const auto& a : Uerr) loc_err[0] = std::max<Real>(loc_err[0], fabs(a));
65 for (const auto& a : Uref) loc_err[1] = std::max<Real>(loc_err[1], fabs(a));
66 comm.Allreduce<Real>(loc_err, glb_err, 2, Comm::CommOp::MAX);
67 if (!comm.Rank()) std::cout<<"Maximum relative error: "<<glb_err[0]/glb_err[1]<<'\n';
68 }
69}

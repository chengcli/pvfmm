10void nbody(vec& sl_coord, vec& sl_den,
11 vec& dl_coord, vec& dl_den,
12 vec& trg_coord, vec& trg_value,
13 const pvfmm::Kernel<double>& kernel_fn, MPI_Comm& comm){
14 int np, rank;
15 MPI_Comm_size(comm, &np);
16 MPI_Comm_rank(comm, &rank);
17
18 long long n_sl = sl_coord.size()/PVFMM_COORD_DIM;
19 long long n_dl = dl_coord.size()/PVFMM_COORD_DIM;
20 long long n_trg_glb=0, n_trg = trg_coord.size()/PVFMM_COORD_DIM;
21 MPI_Allreduce(&n_trg , & n_trg_glb, 1, MPI_LONG_LONG, MPI_SUM, comm);
22
23 vec glb_trg_coord(n_trg_glb*PVFMM_COORD_DIM);
24 vec glb_trg_value(n_trg_glb*kernel_fn.ker_dim[1],0);
25 std::vector<int> recv_disp(np,0);
26 { // Gather all target coordinates.
27 int send_cnt=n_trg*PVFMM_COORD_DIM;
28 std::vector<int> recv_cnts(np);
29 MPI_Allgather(&send_cnt , 1, MPI_INT,
30 &recv_cnts[0], 1, MPI_INT, comm);
31 pvfmm::omp_par::scan(&recv_cnts[0], &recv_disp[0], np);
32 MPI_Allgatherv(&trg_coord[0] , send_cnt , pvfmm::par::Mpi_datatype<double>::value(),
33 &glb_trg_coord[0], &recv_cnts[0], &recv_disp[0], pvfmm::par::Mpi_datatype<double>::value(), comm);
34 }
35
36 { // Evaluate target potential.
37 vec glb_trg_value_(n_trg_glb*kernel_fn.ker_dim[1],0);
38 int omp_p=omp_get_max_threads();
39 #pragma omp parallel for
40 for(int i=0;i<omp_p;i++){
41 size_t a=( i *n_trg_glb)/omp_p;
42 size_t b=((i+1)*n_trg_glb)/omp_p;
43
44 if(kernel_fn.ker_poten!=NULL)
45 kernel_fn.ker_poten(&sl_coord[0], n_sl, &sl_den[0], 1,
46 &glb_trg_coord[0]+a*PVFMM_COORD_DIM, b-a, &glb_trg_value_[0]+a*kernel_fn.ker_dim[1],NULL);
47
48 if(kernel_fn.dbl_layer_poten!=NULL)
49 kernel_fn.dbl_layer_poten(&dl_coord[0], n_dl, &dl_den[0], 1,
50 &glb_trg_coord[0]+a*PVFMM_COORD_DIM, b-a, &glb_trg_value_[0]+a*kernel_fn.ker_dim[1],NULL);
51 }
52 MPI_Allreduce(&glb_trg_value_[0], &glb_trg_value[0], glb_trg_value.size(), pvfmm::par::Mpi_datatype<double>::value(), MPI_SUM, comm);
53 }
54
55 // Get local target values.
56 trg_value.assign(&glb_trg_value[0]+recv_disp[rank]/PVFMM_COORD_DIM*kernel_fn.ker_dim[1], &glb_trg_value[0]+(recv_disp[rank]/PVFMM_COORD_DIM+n_trg)*kernel_fn.ker_dim[1]);
57}

59void fmm_test(size_t N, int mult_order, MPI_Comm comm){
60
61 // Set kernel.
62 const pvfmm::Kernel<double>& kernel_fn=pvfmm::LaplaceKernel<double>::gradient();
63
64 // Create target and source vectors.
65 vec trg_coord=point_distrib<double>(RandUnif,N,comm);
66 vec sl_coord=point_distrib<double>(RandUnif,N,comm);
67 vec dl_coord=point_distrib<double>(RandUnif,0,comm);
68 size_t n_trg = trg_coord.size()/PVFMM_COORD_DIM;
69 size_t n_sl = sl_coord.size()/PVFMM_COORD_DIM;
70 size_t n_dl = dl_coord.size()/PVFMM_COORD_DIM;
71
72 // Set source charges.
73 vec sl_den( n_sl* kernel_fn.ker_dim[0]);
74 vec dl_den(n_dl*(kernel_fn.ker_dim[0]+PVFMM_COORD_DIM));
75 for(size_t i=0;i<sl_den.size();i++) sl_den[i]=drand48();
76 for(size_t i=0;i<dl_den.size();i++) dl_den[i]=drand48();
77
78 // Create memory-manager (optional)
79 pvfmm::mem::MemoryManager mem_mgr(10000000);
80
81 // Construct tree.
82 size_t max_pts=600;
83 auto* tree=PtFMM_CreateTree(sl_coord,sl_den, dl_coord, dl_den, trg_coord, comm, max_pts, pvfmm::FreeSpace);
84
85 // Load matrices.
86 pvfmm::PtFMM<double> matrices(&mem_mgr);
87 matrices.Initialize(mult_order, comm, &kernel_fn);
88
89 // FMM Setup
90 tree->SetupFMM(&matrices);
91
92 // Run FMM
93 vec trg_value;
94 PtFMM_Evaluate(tree, trg_value, n_trg);
95
96 // Re-run FMM
97 tree->ClearFMMData();
98 for(size_t i=0;i<sl_den.size();i++) sl_den[i]=drand48();
99 for(size_t i=0;i<dl_den.size();i++) dl_den[i]=drand48();
100 PtFMM_Evaluate(tree, trg_value, n_trg, &sl_den, &dl_den);
101
102 {// Check error
103 vec trg_sample_coord;
104 vec trg_sample_value;
105 size_t n_trg_sample=0;
106 { // Sample target points for verifications.
107 size_t n_skip=(size_t)(N*n_trg/1e9);
108 if(!n_skip) n_skip=1;
109 for(size_t i=0;i<n_trg;i=i+n_skip){
110 for(size_t j=0;j<PVFMM_COORD_DIM;j++)
111 trg_sample_coord.push_back(trg_coord[i*PVFMM_COORD_DIM+j]);
112 for(int j=0;j<kernel_fn.ker_dim[1];j++)
113 trg_sample_value.push_back(trg_value[i*kernel_fn.ker_dim[1]+j]);
114 n_trg_sample++;
115 }
116 }
117
118 // Direct n-body
119 vec trg_sample_value_(n_trg_sample*kernel_fn.ker_dim[1]);
120 nbody( sl_coord, sl_den ,
121 dl_coord, dl_den ,
122 trg_sample_coord, trg_sample_value_, kernel_fn, comm);
123
124 // Compute error
125 double max_err=0, max_val=0;
126 double max_err_glb=0, max_val_glb=0;
127 for(size_t i=0;i<n_trg_sample*kernel_fn.ker_dim[1];i++){
128 if(fabs(trg_sample_value_[i]-trg_sample_value[i])>max_err)
129 max_err=fabs(trg_sample_value_[i]-trg_sample_value[i]);
130 if(fabs(trg_sample_value_[i])>max_val)
131 max_val=fabs(trg_sample_value_[i]);
132 }
133 MPI_Reduce(&max_err, &max_err_glb, 1, pvfmm::par::Mpi_datatype<double>::value(), MPI_MAX, 0, comm);
134 MPI_Reduce(&max_val, &max_val_glb, 1, pvfmm::par::Mpi_datatype<double>::value(), MPI_MAX, 0, comm);
135
136 int rank;
137 MPI_Comm_rank(comm, &rank);
138 if(!rank) std::cout<<"Maximum Absolute Error:"<<max_err_glb<<'\n';
139 if(!rank) std::cout<<"Maximum Relative Error:"<<max_err_glb/max_val_glb<<'\n';
140 }
141
142 // Free memory
143 delete tree;
144}

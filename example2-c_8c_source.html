7void fn_input(const double* coord, long n, double* out, void* ctx){ //Input function
8 int dof=3;
9 double L=125;
10 for(int i=0;i<n;i++){
11 const double* c=&coord[i*3];
12 {
13 double r_2=(c[0]-0.5)*(c[0]-0.5)+(c[1]-0.5)*(c[1]-0.5)+(c[2]-0.5)*(c[2]-0.5);
14 out[i*dof+0]= 0+2*L*exp(-L*r_2)*(c[0]-0.5);
15 out[i*dof+1]= 4*L*L*(c[2]-0.5)*(5-2*L*r_2)*exp(-L*r_2)+2*L*exp(-L*r_2)*(c[1]-0.5);
16 out[i*dof+2]=-4*L*L*(c[1]-0.5)*(5-2*L*r_2)*exp(-L*r_2)+2*L*exp(-L*r_2)*(c[2]-0.5);
17 }
18 }

34void GetChebNodes(double* cheb_coord, long Nleaf, int cheb_deg, int depth, const double* leaf_coord) {
35 const double leaf_length = 1./(1<<depth);
36 const int Ncheb = (cheb_deg+1)*(cheb_deg+1)*(cheb_deg+1);
37 for (long leaf_idx = 0; leaf_idx < Nleaf; leaf_idx++) {
38 for (long j2 = 0; j2 < cheb_deg+1; j2++) {
39 for (long j1 = 0; j1 < cheb_deg+1; j1++) {
40 for (long j0 = 0; j0 < cheb_deg+1; j0++) {
41 const long node_idx = leaf_idx * Ncheb + (j2 * (cheb_deg+1) + j1) * (cheb_deg+1) + j0;
42 cheb_coord[node_idx*3+0] = leaf_coord[leaf_idx*3+0] + (1-cos(M_PI*(j0*2+1)/(cheb_deg*2+2))) * leaf_length * 0.5;
43 cheb_coord[node_idx*3+1] = leaf_coord[leaf_idx*3+1] + (1-cos(M_PI*(j1*2+1)/(cheb_deg*2+2))) * leaf_length * 0.5;
44 cheb_coord[node_idx*3+2] = leaf_coord[leaf_idx*3+2] + (1-cos(M_PI*(j2*2+1)/(cheb_deg*2+2))) * leaf_length * 0.5;
45 }
46 }
47 }

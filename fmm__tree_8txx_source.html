72template <class FMM_Mat_t>
73void FMM_Tree<FMM_Mat_t>::SetupFMM(FMM_Mat_t* fmm_mat_) {
74 Profile::Tic("SetupFMM",this->Comm(),true);{
75 typedef typename FMM_Mat_t::FMMTree_t MatTree_t;
76 bool device=true;
77
78 #ifdef __INTEL_OFFLOAD
79 Profile::Tic("InitLocks",this->Comm(),false,3);
80 MIC_Lock::init();
81 Profile::Toc();
82 #endif
83
84 //int omp_p=omp_get_max_threads();
85 if(fmm_mat!=fmm_mat_){ // Clear previous setup
86 setup_data.clear();
87 precomp_lst.clear();
88 fmm_mat=fmm_mat_;
89 }
90
91 //Construct LET
92 Profile::Tic("ConstructLET",this->Comm(),false,2);
93 this->ConstructLET(bndry);
94 Profile::Toc();
95
96 //Set Colleagues (Needed to build U, V, W and X lists.)
97 Profile::Tic("SetColleagues",this->Comm(),false,3);
98 this->SetColleagues(bndry);
99 Profile::Toc();
100
101 Profile::Tic("CollectNodeData",this->Comm(),false,3);
102 //Build node list.
103 Node_t* n=dynamic_cast<Node_t*>(this->PostorderFirst());
104 std::vector<Node_t*> all_nodes;
105 while(n!=NULL){
106 n->pt_cnt[0]=0;
107 n->pt_cnt[1]=0;
108 all_nodes.push_back(n);
109 n=static_cast<Node_t*>(this->PostorderNxt(n));
110 }
111 //Collect node data into continuous array.
112 std::vector<Vector<Node_t*> > node_lists; // TODO: Remove this parameter, not really needed
113 fmm_mat->CollectNodeData((MatTree_t*)this,all_nodes, node_data_buff, node_lists);
114 Profile::Toc();
115
116 Profile::Tic("BuildLists",this->Comm(),false,3);
117 BuildInteracLists();
118 Profile::Toc();
119
120 setup_data.resize(8*PVFMM_MAX_DEPTH);
121 precomp_lst.resize(8);
122
123 Profile::Tic("UListSetup",this->Comm(),false,3);
124 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
125 setup_data[i+PVFMM_MAX_DEPTH*0].precomp_data=&precomp_lst[0];
126 fmm_mat->U_ListSetup(setup_data[i+PVFMM_MAX_DEPTH*0],(MatTree_t*)this,node_data_buff,node_lists,fmm_mat->ScaleInvar()?(i==0?-1:PVFMM_MAX_DEPTH+1):i, device);
127 }
128 Profile::Toc();
129 Profile::Tic("WListSetup",this->Comm(),false,3);
130 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
131 setup_data[i+PVFMM_MAX_DEPTH*1].precomp_data=&precomp_lst[1];
132 fmm_mat->W_ListSetup(setup_data[i+PVFMM_MAX_DEPTH*1],(MatTree_t*)this,node_data_buff,node_lists,fmm_mat->ScaleInvar()?(i==0?-1:PVFMM_MAX_DEPTH+1):i, device);
133 }
134 Profile::Toc();
135 Profile::Tic("XListSetup",this->Comm(),false,3);
136 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
137 setup_data[i+PVFMM_MAX_DEPTH*2].precomp_data=&precomp_lst[2];
138 fmm_mat->X_ListSetup(setup_data[i+PVFMM_MAX_DEPTH*2],(MatTree_t*)this,node_data_buff,node_lists,fmm_mat->ScaleInvar()?(i==0?-1:PVFMM_MAX_DEPTH+1):i, device);
139 }
140 Profile::Toc();
141 Profile::Tic("VListSetup",this->Comm(),false,3);
142 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
143 setup_data[i+PVFMM_MAX_DEPTH*3].precomp_data=&precomp_lst[3];
144 fmm_mat->V_ListSetup(setup_data[i+PVFMM_MAX_DEPTH*3],(MatTree_t*)this,node_data_buff,node_lists,fmm_mat->ScaleInvar()?(i==0?-1:PVFMM_MAX_DEPTH+1):i, /*device*/ false);
145 }
146 Profile::Toc();
147 Profile::Tic("D2DSetup",this->Comm(),false,3);
148 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
149 setup_data[i+PVFMM_MAX_DEPTH*4].precomp_data=&precomp_lst[4];
150 fmm_mat->Down2DownSetup(setup_data[i+PVFMM_MAX_DEPTH*4],(MatTree_t*)this,node_data_buff,node_lists,i, /*device*/ false);
151 }
152 Profile::Toc();
153 Profile::Tic("D2TSetup",this->Comm(),false,3);
154 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
155 setup_data[i+PVFMM_MAX_DEPTH*5].precomp_data=&precomp_lst[5];
156 fmm_mat->Down2TargetSetup(setup_data[i+PVFMM_MAX_DEPTH*5],(MatTree_t*)this,node_data_buff,node_lists,fmm_mat->ScaleInvar()?(i==0?-1:PVFMM_MAX_DEPTH+1):i, /*device*/ false);
157 }
158 Profile::Toc();
159
160 Profile::Tic("S2USetup",this->Comm(),false,3);
161 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
162 setup_data[i+PVFMM_MAX_DEPTH*6].precomp_data=&precomp_lst[6];
163 fmm_mat->Source2UpSetup(setup_data[i+PVFMM_MAX_DEPTH*6],(MatTree_t*)this,node_data_buff,node_lists,fmm_mat->ScaleInvar()?(i==0?-1:PVFMM_MAX_DEPTH+1):i, /*device*/ false);
164 }
165 Profile::Toc();
166 Profile::Tic("U2USetup",this->Comm(),false,3);
167 for(size_t i=0;i<PVFMM_MAX_DEPTH;i++){
168 setup_data[i+PVFMM_MAX_DEPTH*7].precomp_data=&precomp_lst[7];
169 fmm_mat->Up2UpSetup(setup_data[i+PVFMM_MAX_DEPTH*7],(MatTree_t*)this,node_data_buff,node_lists,i, /*device*/ false);
170 }
171 Profile::Toc();
172
173 #ifdef __INTEL_OFFLOAD
174 int wait_lock_idx=-1;
175 wait_lock_idx=MIC_Lock::curr_lock();
176 #pragma offload target(mic:0)
177 {MIC_Lock::wait_lock(wait_lock_idx);}
178 #endif
179
180 ClearFMMData();
181
182 }Profile::Toc();

189 bool device=true;

210 if(mat && mat->Dim(0)*mat->Dim(1)>0){
211 size_t a=(mat->Dim(0)*mat->Dim(1)*(j+0))/omp_p;

227 {MIC_Lock::wait_lock(wait_lock_idx);}

231 }

233 }Profile::Toc();

267template <class FMM_Mat_t>
268void FMM_Tree<FMM_Mat_t>::UpwardPass() {
269 //bool device=true;
270
271 int max_depth=0;
272 { // Get max_depth
273 int max_depth_loc=0;
274 std::vector<Node_t*>& nodes=this->GetNodeList();
275 for(size_t i=0;i<nodes.size();i++){
276 Node_t* n=nodes[i];
277 if(n->Depth()>max_depth_loc) max_depth_loc=n->Depth();
278 }
279 MPI_Allreduce(&max_depth_loc, &max_depth, 1, MPI_INT, MPI_MAX, *this->Comm());
280 }
281
282 //Upward Pass (initialize all leaf nodes)
283 Profile::Tic("S2U",this->Comm(),false,5);
284 for(int i=0; i<=(fmm_mat->ScaleInvar()?0:max_depth); i++){ // Source2Up
285 if(!fmm_mat->ScaleInvar()) fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*6],/*device*/ false);
286 fmm_mat->Source2Up(setup_data[i+PVFMM_MAX_DEPTH*6]);
287 }
288 Profile::Toc();
289
290 //Upward Pass (level by level)
291 Profile::Tic("U2U",this->Comm(),false,5);
292 for(int i=max_depth-1; i>=0; i--){ // Up2Up
293 if(!fmm_mat->ScaleInvar()) fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*7],/*device*/ false);
294 fmm_mat->Up2Up(setup_data[i+PVFMM_MAX_DEPTH*7]);
295 }
296 Profile::Toc();
297}

300template <class FMM_Mat_t>
301void FMM_Tree<FMM_Mat_t>::BuildInteracLists() {
302 std::vector<Node_t*> n_list_src;
303 std::vector<Node_t*> n_list_trg;
304 { // Build n_list
305 std::vector<Node_t*>& nodes=this->GetNodeList();
306 for(size_t i=0;i<nodes.size();i++){
307 if(!nodes[i]->IsGhost() && nodes[i]->pt_cnt[0]){
308 n_list_src.push_back(nodes[i]);
309 }
310 if(!nodes[i]->IsGhost() && nodes[i]->pt_cnt[1]){
311 n_list_trg.push_back(nodes[i]);
312 }
313 }
314 }
315 size_t node_cnt=std::max(n_list_src.size(),n_list_trg.size());
316
317 std::vector<Mat_Type> type_lst;
318 std::vector<std::vector<Node_t*>*> type_node_lst;
319 type_lst.push_back(S2U_Type); type_node_lst.push_back(&n_list_src);
320 type_lst.push_back(U2U_Type); type_node_lst.push_back(&n_list_src);
321 type_lst.push_back(D2D_Type); type_node_lst.push_back(&n_list_trg);
322 type_lst.push_back(D2T_Type); type_node_lst.push_back(&n_list_trg);
323 type_lst.push_back(U0_Type ); type_node_lst.push_back(&n_list_trg);
324 type_lst.push_back(U1_Type ); type_node_lst.push_back(&n_list_trg);
325 type_lst.push_back(U2_Type ); type_node_lst.push_back(&n_list_trg);
326 type_lst.push_back(W_Type ); type_node_lst.push_back(&n_list_trg);
327 type_lst.push_back(X_Type ); type_node_lst.push_back(&n_list_trg);
328 type_lst.push_back(V1_Type ); type_node_lst.push_back(&n_list_trg);
329 std::vector<size_t> interac_cnt(type_lst.size());
330 std::vector<size_t> interac_dsp(type_lst.size(),0);
331 for(size_t i=0;i<type_lst.size();i++){
332 interac_cnt[i]=interac_list.ListCount(type_lst[i]);
333 }
334 omp_par::scan(&interac_cnt[0],&interac_dsp[0],type_lst.size());
335 node_interac_lst.ReInit(node_cnt,interac_cnt.back()+interac_dsp.back());
336
337 // Build interaction lists.
338 int omp_p=omp_get_max_threads();
339 #pragma omp parallel for
340 for(int j=0;j<omp_p;j++){
341 for(size_t k=0;k<type_lst.size();k++){
342 std::vector<Node_t*>& n_list=*type_node_lst[k];
343 size_t a=(n_list.size()*(j ))/omp_p;
344 size_t b=(n_list.size()*(j+1))/omp_p;
345 for(size_t i=a;i<b;i++){
346 Node_t* n=n_list[i];
347 n->interac_list[type_lst[k]].ReInit(interac_cnt[k],&node_interac_lst[i][interac_dsp[k]],false);
348 interac_list.BuildList(n,type_lst[k]);
349 }
350 }
351 }
352}

355template <class FMM_Mat_t>
356void FMM_Tree<FMM_Mat_t>::MultipoleReduceBcast() {
357 int num_p,rank;
358 MPI_Comm_size(*this->Comm(),&num_p);
359 MPI_Comm_rank(*this->Comm(),&rank );
360 if(num_p==1) return;
361
362 Profile::Tic("Reduce",this->Comm(),true,3);
363 std::vector<MortonId> mins=this->GetMins();
364
365 size_t bit_mask=1;
366 size_t max_child=(1UL<<this->Dim());
367
368 //Initialize initial send nodes.
369 std::vector<Node_t*> send_nodes[2];
370
371 //Initialize send_node[0]
372 Node_t* tmp_node=static_cast<Node_t*>(this->RootNode());
373 assert(!tmp_node->IsGhost());
374 while(!tmp_node->IsLeaf()){
375 Node_t* tmp_node_=NULL;
376 for(size_t i=0;i<max_child;i++){
377 tmp_node_=static_cast<Node_t*>(tmp_node->Child(i));
378 if(tmp_node_!=NULL) if(!tmp_node_->IsGhost()) break;
379 }
380 tmp_node=tmp_node_; assert(tmp_node!=NULL);
381 }
382 int n[2];
383 n[0]=tmp_node->Depth()+1;
384 send_nodes[0].resize(n[0]);
385 send_nodes[0][n[0]-1]=tmp_node;
386 for(int i=n[0]-1;i>0;i--)
387 send_nodes[0][i-1]=static_cast<Node_t*>(send_nodes[0][i]->Parent());
388
389 //Initialize send_node[1]
390 tmp_node=static_cast<Node_t*>(this->RootNode());
391 while(!tmp_node->IsLeaf()){
392 Node_t* tmp_node_=NULL;
393 for(int i=max_child-1;i>=0;i--){
394 tmp_node_=static_cast<Node_t*>(tmp_node->Child(i));
395 if(tmp_node_!=NULL) if(!tmp_node_->IsGhost()) break;
396 }
397 tmp_node=tmp_node_; assert(tmp_node!=NULL);
398 }
399 n[1]=tmp_node->Depth()+1;
400 send_nodes[1].resize(n[1]);
401 send_nodes[1][n[1]-1]=tmp_node;
402 for(int i=n[1]-1;i>0;i--)
403 send_nodes[1][i-1]=static_cast<Node_t*>(send_nodes[1][i]->Parent());
404
405 //Hypercube reduction.
406 while(bit_mask<(size_t)num_p){
407 size_t partner=rank^bit_mask; //Partner process id
408 int merge_indx=((bit_mask & rank)==0?1:0);
409 bit_mask=bit_mask<<1;
410 //if(rank >= num_p - (num_p % bit_mask)) break;
411
412 //Initialize send data.
413 size_t s_node_cnt[2]={send_nodes[0].size(),send_nodes[1].size()};
414 int send_size=2*sizeof(size_t)+(s_node_cnt[0]+s_node_cnt[1])*sizeof(MortonId);
415 std::vector<PackedData> send_data(s_node_cnt[0]+s_node_cnt[1]);
416
417 size_t s_iter=0;
418 for(int i=0;i<2;i++)
419 for(size_t j=0;j<s_node_cnt[i];j++){
420 assert(send_nodes[i][j]!=NULL);
421 send_data[s_iter]=send_nodes[i][j]->PackMultipole();
422 send_size+=send_data[s_iter].length+sizeof(size_t);
423 s_iter++;
424 }
425
426 char* send_buff=mem::aligned_new<char>(send_size);
427 char* buff_iter=send_buff;
428 ((size_t*)buff_iter)[0]=s_node_cnt[0];
429 ((size_t*)buff_iter)[1]=s_node_cnt[1];
430 buff_iter+=2*sizeof(size_t);
431
432 s_iter=0;
433 for(int i=0;i<2;i++)
434 for(size_t j=0;j<s_node_cnt[i];j++){
435 ((MortonId*)buff_iter)[0]=send_nodes[i][j]->GetMortonId();
436 buff_iter+=sizeof(MortonId);
437
438 ((size_t*)buff_iter)[0]=send_data[s_iter].length;
439 buff_iter+=sizeof(size_t);
440
441 mem::copy<char>((char*)buff_iter,(char*)send_data[s_iter].data,send_data[s_iter].length);
442 buff_iter+=send_data[s_iter].length;
443
444 s_iter++;
445 }
446
447 //Exchange send and recv sizes
448 int recv_size=0;
449 MPI_Status status;
450 char* recv_buff=NULL;
451 if(partner<(size_t)num_p){
452 MPI_Sendrecv(&send_size, 1, MPI_INT, partner, 0, &recv_size, 1, MPI_INT, partner, 0, *this->Comm(), &status);
453 recv_buff=mem::aligned_new<char>(recv_size);
454 MPI_Sendrecv(send_buff, send_size, MPI_BYTE, partner, 0, recv_buff, recv_size, MPI_BYTE, partner, 0, *this->Comm(), &status);
455 }
456
457 //Need an extra broadcast for incomplete hypercubes.
458 size_t p0_start=num_p - (num_p % (bit_mask ));
459 size_t p0_end =num_p - (num_p % (bit_mask>>1));
460 if(((size_t)rank >= p0_start) && ((size_t)num_p>p0_end) && ((size_t)rank < p0_end) ){
461 size_t bit_mask0=1;
462 size_t num_p0=p0_end-p0_start;
463 while( bit_mask0 < num_p0 ){
464 if( (bit_mask0<<1) > (num_p - p0_end) ){
465 size_t partner0=rank^bit_mask0;
466 if( rank-p0_start < bit_mask0 ){
467 //Send
468 MPI_Send(&recv_size, 1, MPI_INT , partner0, 0, *this->Comm());
469 MPI_Send( recv_buff, recv_size, MPI_BYTE, partner0, 0, *this->Comm());
470 }else if( rank-p0_start < (bit_mask0<<1) ){
471 //Receive
472 if(recv_size>0) mem::aligned_delete<char>(recv_buff);
473 MPI_Recv(&recv_size, 1, MPI_INT , partner0, 0, *this->Comm(), &status);
474 recv_buff=mem::aligned_new<char>(recv_size);
475 MPI_Recv( recv_buff, recv_size, MPI_BYTE, partner0, 0, *this->Comm(), &status);
476 }
477 }
478 bit_mask0=bit_mask0<<1;
479 }
480 }
481
482 //Construct nodes from received data.
483 if(recv_size>0){
484 buff_iter=recv_buff;
485 size_t r_node_cnt[2]={((size_t*)buff_iter)[0],((size_t*)buff_iter)[1]};
486 buff_iter+=2*sizeof(size_t);
487 std::vector<MortonId> r_mid[2];
488 r_mid[0].resize(r_node_cnt[0]);
489 r_mid[1].resize(r_node_cnt[1]);
490 std::vector<Node_t*> recv_nodes[2];
491 recv_nodes[0].resize(r_node_cnt[0]);
492 recv_nodes[1].resize(r_node_cnt[1]);
493 std::vector<PackedData> recv_data[2];
494 recv_data[0].resize(r_node_cnt[0]);
495 recv_data[1].resize(r_node_cnt[1]);
496 for(int i=0;i<2;i++)
497 for(size_t j=0;j<r_node_cnt[i];j++){
498 r_mid[i][j]=((MortonId*)buff_iter)[0];
499 buff_iter+=sizeof(MortonId);
500
501 recv_data[i][j].length=((size_t*)buff_iter)[0];
502 buff_iter+=sizeof(size_t);
503
504 recv_data[i][j].data=(void*)buff_iter;
505 buff_iter+=recv_data[i][j].length;
506 }
507
508 // Add multipole expansion to existing nodes.
509 for(size_t i=0;i<r_node_cnt[1-merge_indx];i++){
510 if(i<send_nodes[merge_indx].size()){
511 if(r_mid[1-merge_indx][i]==send_nodes[merge_indx][i]->GetMortonId()){
512 send_nodes[merge_indx][i]->AddMultipole(recv_data[1-merge_indx][i]);
513 }else break;
514 }else break;
515 }
516
517 bool new_branch=false;
518 for(size_t i=0;i<r_node_cnt[merge_indx];i++){
519 if(i<send_nodes[merge_indx].size() && !new_branch){
520 if(r_mid[merge_indx][i]==send_nodes[merge_indx][i]->GetMortonId()){
521 recv_nodes[merge_indx][i]=send_nodes[merge_indx][i];
522 }else{
523 new_branch=true;
524 size_t n_=(i<(size_t)n[merge_indx]?n[merge_indx]:i);
525 for(size_t j=n_;j<send_nodes[merge_indx].size();j++)
526 mem::aligned_delete(send_nodes[merge_indx][j]);
527 if(i<(size_t)n[merge_indx]) n[merge_indx]=i;
528 }
529 }
530 if(i>=send_nodes[merge_indx].size() || new_branch){
531 recv_nodes[merge_indx][i]=static_cast<Node_t*>(this->NewNode());
532 recv_nodes[merge_indx][i]->SetCoord(r_mid[merge_indx][i]);
533 recv_nodes[merge_indx][i]->InitMultipole(recv_data[merge_indx][i]);
534 }
535 }
536 send_nodes[merge_indx]=recv_nodes[merge_indx];
537 }
538 mem::aligned_delete<char>(send_buff);
539 mem::aligned_delete<char>(recv_buff);
540 }
541
542 for(int i=0;i<2;i++)
543 for(size_t j=n[i];j<send_nodes[i].size();j++)
544 mem::aligned_delete(send_nodes[i][j]);
545 Profile::Toc();
546
547 //Now Broadcast nodes to build LET.
548 Profile::Tic("Broadcast",this->Comm(),true,4);
549 this->ConstructLET(bndry);
550 Profile::Toc();
551}

554template <class FMM_Mat_t>
555void FMM_Tree<FMM_Mat_t>::DownwardPass() {
556 bool device=true;
557
558 Profile::Tic("Setup",this->Comm(),true,3);
559 std::vector<Node_t*> leaf_nodes;
560 int max_depth=0;
561 { // Build leaf node list
562 int max_depth_loc=0;
563 std::vector<Node_t*>& nodes=this->GetNodeList();
564 for(size_t i=0;i<nodes.size();i++){
565 Node_t* n=nodes[i];
566 if(!n->IsGhost() && n->IsLeaf()) leaf_nodes.push_back(n);
567 if(n->Depth()>max_depth_loc) max_depth_loc=n->Depth();
568 }
569 MPI_Allreduce(&max_depth_loc, &max_depth, 1, MPI_INT, MPI_MAX, *this->Comm());
570 }
571 Profile::Toc();
572
573 #if defined(__INTEL_OFFLOAD) || defined(PVFMM_HAVE_CUDA)
574 if(device){ // Host2Device:Src
575 Profile::Tic("Host2Device:Src",this->Comm(),false,5);
576 if(setup_data[0+PVFMM_MAX_DEPTH*2]. coord_data!=NULL) setup_data[0+PVFMM_MAX_DEPTH*2]. coord_data->AllocDevice(true);
577 if(setup_data[0+PVFMM_MAX_DEPTH*2]. input_data!=NULL) setup_data[0+PVFMM_MAX_DEPTH*2]. input_data->AllocDevice(true);
578 Profile::Toc();
579 }
580 #endif
581
582 Profile::Tic("BoundaryCondition",this->Comm(),false,5);
583 fmm_mat->PeriodicBC(dynamic_cast<Node_t*>(this->RootNode()), bndry);
584 Profile::Toc();
585
586 for(int i=0; i<=(fmm_mat->ScaleInvar()?0:max_depth); i++){ // U,V,W,X-lists
587
588 if(!fmm_mat->ScaleInvar()){ // Precomp
589 std::stringstream level_str;
590 level_str<<"Level-"<<std::setfill('0')<<std::setw(2)<<i<<"\0";
591 Profile::Tic(level_str.str().c_str(),this->Comm(),false,5);
592
593 Profile::Tic("Precomp",this->Comm(),false,5);
594 {// Precomp U
595 Profile::Tic("Precomp-U",this->Comm(),false,10);
596 fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*0],device);
597 Profile::Toc();
598 }
599 {// Precomp W
600 Profile::Tic("Precomp-W",this->Comm(),false,10);
601 fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*1],device);
602 Profile::Toc();
603 }
604 {// Precomp X
605 Profile::Tic("Precomp-X",this->Comm(),false,10);
606 fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*2],device);
607 Profile::Toc();
608 }
609 if(0){// Precomp V
610 Profile::Tic("Precomp-V",this->Comm(),false,10);
611 fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*3], /*device*/ false);
612 Profile::Toc();
613 }
614 Profile::Toc();
615 }
616
617 {// X-List
618 Profile::Tic("X-List",this->Comm(),false,5);
619 fmm_mat->X_List(setup_data[i+PVFMM_MAX_DEPTH*2], device);
620 Profile::Toc();
621 }
622
623 #if defined(__INTEL_OFFLOAD) || defined(PVFMM_HAVE_CUDA)
624 if(i==0 && device){ // Host2Device:Mult
625 Profile::Tic("Host2Device:Mult",this->Comm(),false,5);
626 if(setup_data[0+PVFMM_MAX_DEPTH*1]. input_data!=NULL) setup_data[0+PVFMM_MAX_DEPTH*1]. input_data->AllocDevice(true);
627 Profile::Toc();
628 }
629
630 if(device) if(i==(fmm_mat->ScaleInvar()?0:max_depth)){ // Device2Host: LocalExp
631 Profile::Tic("Device2Host:LocExp",this->Comm(),false,5);
632 if(setup_data[0+PVFMM_MAX_DEPTH*2].output_data!=NULL){
633 Matrix<Real_t>& output_data=*setup_data[0+PVFMM_MAX_DEPTH*2].output_data;
634 if(fmm_mat->staging_buffer.Dim()){
635 assert(fmm_mat->staging_buffer.Dim()*sizeof(Real_t)>=output_data.Dim(0)*output_data.Dim(1));
636 output_data.Device2Host((Real_t*)&fmm_mat->staging_buffer[0]);
637 }
638 }
639 Profile::Toc();
640 }
641 #endif
642
643 {// W-List
644 Profile::Tic("W-List",this->Comm(),false,5);
645 fmm_mat->W_List(setup_data[i+PVFMM_MAX_DEPTH*1], device);
646 Profile::Toc();
647 }
648
649 {// U-List
650 Profile::Tic("U-List",this->Comm(),false,5);
651 fmm_mat->U_List(setup_data[i+PVFMM_MAX_DEPTH*0], device);
652 Profile::Toc();
653 }
654
655 {// V-List
656 Profile::Tic("V-List",this->Comm(),false,5);
657 fmm_mat->V_List(setup_data[i+PVFMM_MAX_DEPTH*3], /*device*/ false);
658 Profile::Toc();
659 }
660
661 if(!fmm_mat->ScaleInvar()){ // Wait
662 #ifdef __INTEL_OFFLOAD
663 int wait_lock_idx=-1;
664 if(device) wait_lock_idx=MIC_Lock::curr_lock();
665 #pragma offload if(device) target(mic:0)
666 {if(device) MIC_Lock::wait_lock(wait_lock_idx);}
667 #endif
668 Profile::Toc();
669 }
670 }
671
672 #if defined(__INTEL_OFFLOAD) || defined(PVFMM_HAVE_CUDA)
673 Profile::Tic("D2H_Wait:LocExp",this->Comm(),false,5);
674 if(device) if(setup_data[0+PVFMM_MAX_DEPTH*2].output_data!=NULL){
675 Real_t* dev_ptr=(Real_t*)&fmm_mat->staging_buffer[0];
676 Matrix<Real_t>& output_data=*setup_data[0+PVFMM_MAX_DEPTH*2].output_data;
677 size_t n=output_data.Dim(0)*output_data.Dim(1);
678 if(fmm_mat->staging_buffer.Dim()){
679 Real_t* host_ptr=output_data[0];
680 output_data.Device2HostWait();
681
682 #pragma omp parallel for
683 for(size_t i=0;i<n;i++){
684 host_ptr[i]+=dev_ptr[i];
685 }
686 }
687 }
688 Profile::Toc();
689
690 Profile::Tic("Device2Host:Trg",this->Comm(),false,5);
691 if(device) if(setup_data[0+PVFMM_MAX_DEPTH*0].output_data!=NULL){ // Device2Host: Target
692 Matrix<Real_t>& output_data=*setup_data[0+PVFMM_MAX_DEPTH*0].output_data;
693 if(fmm_mat->staging_buffer.Dim()){
694 assert(fmm_mat->staging_buffer.Dim()>=sizeof(Real_t)*output_data.Dim(0)*output_data.Dim(1));
695 output_data.Device2Host((Real_t*)&fmm_mat->staging_buffer[0]);
696 }
697 }
698 Profile::Toc();
699 #endif
700
701 Profile::Tic("D2D",this->Comm(),false,5);
702 for(int i=0; i<=max_depth; i++){ // Down2Down
703 if(!fmm_mat->ScaleInvar()) fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*4],/*device*/ false);
704 fmm_mat->Down2Down(setup_data[i+PVFMM_MAX_DEPTH*4]);
705 }
706 Profile::Toc();
707
708 Profile::Tic("D2T",this->Comm(),false,5);
709 for(int i=0; i<=(fmm_mat->ScaleInvar()?0:max_depth); i++){ // Down2Target
710 if(!fmm_mat->ScaleInvar()) fmm_mat->SetupPrecomp(setup_data[i+PVFMM_MAX_DEPTH*5],/*device*/ false);
711 fmm_mat->Down2Target(setup_data[i+PVFMM_MAX_DEPTH*5]);
712 }
713 Profile::Toc();
714
715 #if defined(__INTEL_OFFLOAD) || defined(PVFMM_HAVE_CUDA)
716 Profile::Tic("D2H_Wait:Trg",this->Comm(),false,5);
717 if(device) if(setup_data[0+PVFMM_MAX_DEPTH*0].output_data!=NULL){
718 Real_t* dev_ptr=(Real_t*)&fmm_mat->staging_buffer[0];
719 Matrix<Real_t>& output_data=*setup_data[0+PVFMM_MAX_DEPTH*0].output_data;
720 size_t n=output_data.Dim(0)*output_data.Dim(1);
721 if(fmm_mat->staging_buffer.Dim()){
722 Real_t* host_ptr=output_data[0];
723 output_data.Device2HostWait();
724
725 #pragma omp parallel for
726 for(size_t i=0;i<n;i++){
727 host_ptr[i]+=dev_ptr[i];
728 }
729 }
730 }
731 Profile::Toc();
732 #endif
733
734 Profile::Tic("PostProc",this->Comm(),false,5);
735 typedef typename FMM_Mat_t::FMMTree_t MatTree_t;
736 fmm_mat->PostProcessing((MatTree_t*)this, leaf_nodes, bndry);
737 Profile::Toc();
738}

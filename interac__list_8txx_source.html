95/**
96 * \brief Build interaction list for this node.
97 */
98template <class Node_t>
99void InteracList<Node_t>::BuildList(Node_t* n, Mat_Type t){
100 Vector<Node_t*>& interac_list=n->interac_list[t];
101 if(interac_list.Dim()!=ListCount(t)) interac_list.ReInit(ListCount(t));
102 interac_list.SetZero();
103
104 static const int n_collg=sctl::pow<unsigned int>(3,dim);
105 static const int n_child=sctl::pow<unsigned int>(2,dim);
106 int rel_coord[3];
107
108 switch (t){
109
110 case S2U_Type:
111 {
112 if(!n->IsGhost() && n->IsLeaf()) interac_list[0]=n;
113 break;
114 }
115 case U2U_Type:
116 {
117 if(n->IsGhost() || n->IsLeaf()) return;
118 for(int j=0;j<n_child;j++){
119 rel_coord[0]=-1+(j & 1?2:0);
120 rel_coord[1]=-1+(j & 2?2:0);
121 rel_coord[2]=-1+(j & 4?2:0);
122 int c_hash = coord_hash(rel_coord);
123 int idx=hash_lut[t][c_hash];
124 Node_t* chld=(Node_t*)n->Child(j);
125 if(idx>=0 && !chld->IsGhost()) interac_list[idx]=chld;
126 }
127 break;
128 }
129 case D2D_Type:
130 {
131 if(n->IsGhost() || n->Parent()==NULL) return;
132 Node_t* p=(Node_t*)n->Parent();
133 int p2n=n->Path2Node();
134 {
135 rel_coord[0]=-1+(p2n & 1?2:0);
136 rel_coord[1]=-1+(p2n & 2?2:0);
137 rel_coord[2]=-1+(p2n & 4?2:0);
138 int c_hash = coord_hash(rel_coord);
139 int idx=hash_lut[t][c_hash];
140 if(idx>=0) interac_list[idx]=p;
141 }
142 break;
143 }
144 case D2T_Type:
145 {
146 if(!n->IsGhost() && n->IsLeaf()) interac_list[0]=n;
147 break;
148 }
149 case U0_Type:
150 {
151 if(n->IsGhost() || n->Parent()==NULL || !n->IsLeaf()) return;
152 Node_t* p=(Node_t*)n->Parent();
153 int p2n=n->Path2Node();
154 for(int i=0;i<n_collg;i++){
155 Node_t* pc=(Node_t*)p->Colleague(i);
156 if(pc!=NULL && pc->IsLeaf()){
157 rel_coord[0]=( i %3)*4-4-(p2n & 1?2:0)+1;
158 rel_coord[1]=((i/3)%3)*4-4-(p2n & 2?2:0)+1;
159 rel_coord[2]=((i/9)%3)*4-4-(p2n & 4?2:0)+1;
160 int c_hash = coord_hash(rel_coord);
161 int idx=hash_lut[t][c_hash];
162 if(idx>=0) interac_list[idx]=pc;
163 }
164 }
165 break;
166 }
167 case U1_Type:
168 {
169 if(n->IsGhost() || !n->IsLeaf()) return;
170 for(int i=0;i<n_collg;i++){
171 Node_t* col=(Node_t*)n->Colleague(i);
172 if(col!=NULL && col->IsLeaf()){
173 rel_coord[0]=( i %3)-1;
174 rel_coord[1]=((i/3)%3)-1;
175 rel_coord[2]=((i/9)%3)-1;
176 int c_hash = coord_hash(rel_coord);
177 int idx=hash_lut[t][c_hash];
178 if(idx>=0) interac_list[idx]=col;
179 }
180 }
181 break;
182 }
183 case U2_Type:
184 {
185 if(n->IsGhost() || !n->IsLeaf()) return;
186 for(int i=0;i<n_collg;i++){
187 Node_t* col=(Node_t*)n->Colleague(i);
188 if(col!=NULL && !col->IsLeaf()){
189 for(int j=0;j<n_child;j++){
190 rel_coord[0]=( i %3)*4-4+(j & 1?2:0)-1;
191 rel_coord[1]=((i/3)%3)*4-4+(j & 2?2:0)-1;
192 rel_coord[2]=((i/9)%3)*4-4+(j & 4?2:0)-1;
193 int c_hash = coord_hash(rel_coord);
194 int idx=hash_lut[t][c_hash];
195 if(idx>=0){
196 assert(col->Child(j)->IsLeaf()); //2:1 balanced
197 interac_list[idx]=(Node_t*)col->Child(j);
198 }
199 }
200 }
201 }
202 break;
203 }
204 case V_Type:
205 {
206 if(n->IsGhost() || n->Parent()==NULL) return;
207 Node_t* p=(Node_t*)n->Parent();
208 int p2n=n->Path2Node();
209 for(int i=0;i<n_collg;i++){
210 Node_t* pc=(Node_t*)p->Colleague(i);
211 if(pc!=NULL?!pc->IsLeaf():0){
212 for(int j=0;j<n_child;j++){
213 rel_coord[0]=( i %3)*2-2+(j & 1?1:0)-(p2n & 1?1:0);
214 rel_coord[1]=((i/3)%3)*2-2+(j & 2?1:0)-(p2n & 2?1:0);
215 rel_coord[2]=((i/9)%3)*2-2+(j & 4?1:0)-(p2n & 4?1:0);
216 int c_hash = coord_hash(rel_coord);
217 int idx=hash_lut[t][c_hash];
218 if(idx>=0) interac_list[idx]=(Node_t*)pc->Child(j);
219 }
220 }
221 }
222 break;
223 }
224 case V1_Type:
225 {
226 if(n->IsGhost() || n->IsLeaf()) return;
227 for(int i=0;i<n_collg;i++){
228 Node_t* col=(Node_t*)n->Colleague(i);
229 if(col!=NULL && !col->IsLeaf()){
230 rel_coord[0]=( i %3)-1;
231 rel_coord[1]=((i/3)%3)-1;
232 rel_coord[2]=((i/9)%3)-1;
233 int c_hash = coord_hash(rel_coord);
234 int idx=hash_lut[t][c_hash];
235 if(idx>=0) interac_list[idx]=col;
236 }
237 }
238 break;
239 }
240 case W_Type:
241 {
242 if(n->IsGhost() || !n->IsLeaf()) return;
243 for(int i=0;i<n_collg;i++){
244 Node_t* col=(Node_t*)n->Colleague(i);
245 if(col!=NULL && !col->IsLeaf()){
246 for(int j=0;j<n_child;j++){
247 rel_coord[0]=( i %3)*4-4+(j & 1?2:0)-1;
248 rel_coord[1]=((i/3)%3)*4-4+(j & 2?2:0)-1;
249 rel_coord[2]=((i/9)%3)*4-4+(j & 4?2:0)-1;
250 int c_hash = coord_hash(rel_coord);
251 int idx=hash_lut[t][c_hash];
252 if(idx>=0) interac_list[idx]=(Node_t*)col->Child(j);
253 }
254 }
255 }
256 break;
257 }
258 case X_Type:
259 {
260 if(n->IsGhost() || n->Parent()==NULL) return;
261 Node_t* p=(Node_t*)n->Parent();
262 int p2n=n->Path2Node();
263 for(int i=0;i<n_collg;i++){
264 Node_t* pc=(Node_t*)p->Colleague(i);
265 if(pc!=NULL && pc->IsLeaf()){
266 rel_coord[0]=( i %3)*4-4-(p2n & 1?2:0)+1;
267 rel_coord[1]=((i/3)%3)*4-4-(p2n & 2?2:0)+1;
268 rel_coord[2]=((i/9)%3)*4-4-(p2n & 4?2:0)+1;
269 int c_hash = coord_hash(rel_coord);
270 int idx=hash_lut[t][c_hash];
271 if(idx>=0) interac_list[idx]=pc;
272 }
273 }
274 break;
275 }
276 default:
277 break;
278 }
279}

287template <class Node_t>
288Permutation<typename Node_t::Real_t>& InteracList<Node_t>::Perm_R(int l, Mat_Type type, size_t indx){
289 assert(l>=0);
290 size_t indx0=InteracClass(type, indx);
291 Matrix <Real_t>& M0 =mat->Mat (l, type, indx0);
292 Permutation<Real_t>& row_perm=mat->Perm_R(l, type, indx );
293 if(M0.Dim(0)==0 || M0.Dim(1)==0) return row_perm;
294
295 //Get the necessary permutations.
296 if(row_perm.Dim()==0){
297 std::vector<Perm_Type> p_list=PermutList(type, indx);
298 for(int i=0;i<l;i++) p_list.push_back(Scaling);
299 Permutation<Real_t> row_perm_=Permutation<Real_t>(M0.Dim(0));
300 for(int i=0;i<C_Perm;i++){
301 Permutation<Real_t>& pr=mat->Perm(type, R_Perm + i);
302 if(!pr.Dim()) row_perm_=Permutation<Real_t>(0);
303 }
304
305 if(row_perm_.Dim()>0)
306 for(int i=p_list.size()-1; i>=0; i--){
307 assert(type!=V_Type);
308 Permutation<Real_t>& pr=mat->Perm(type, R_Perm + p_list[i]);
309 row_perm_=pr.Transpose()*row_perm_;
310 }
311 row_perm=row_perm_;
312 }
313 return row_perm;
314}

316template <class Node_t>
317Permutation<typename Node_t::Real_t>& InteracList<Node_t>::Perm_C(int l, Mat_Type type, size_t indx){
318 assert(l>=0);
319 size_t indx0=InteracClass(type, indx);
320 Matrix <Real_t>& M0 =mat->Mat (l, type, indx0);
321 Permutation<Real_t>& col_perm=mat->Perm_C(l, type, indx );
322 if(M0.Dim(0)==0 || M0.Dim(1)==0) return col_perm;
323
324 //Get the necessary permutations.
325 if(col_perm.Dim()==0){
326 std::vector<Perm_Type> p_list=PermutList(type, indx);
327 for(int i=0;i<l;i++) p_list.push_back(Scaling);
328 Permutation<Real_t> col_perm_=Permutation<Real_t>(M0.Dim(1));
329 for(int i=0;i<C_Perm;i++){
330 Permutation<Real_t>& pc=mat->Perm(type, C_Perm + i);
331 if(!pc.Dim()) col_perm_=Permutation<Real_t>(0);
332 }
333
334 if(col_perm_.Dim()>0)
335 for(int i=p_list.size()-1; i>=0; i--){
336 assert(type!=V_Type);
337 Permutation<Real_t>& pc=mat->Perm(type, C_Perm + p_list[i]);
338 col_perm_=col_perm_*pc;
339 }
340 col_perm=col_perm_;
341 }
342 return col_perm;
343}

345/**
346 * \brief A hash function defined on the relative coordinates of octants.
347 */
348#define PVFMM_MAX_COORD_HASH 2000
349template <class Node_t>
350int InteracList<Node_t>::coord_hash(int* c){
351 const int n=5;
352 return ( (c[2]+n) * (2*n) + (c[1]+n) ) *(2*n) + (c[0]+n);
353}
354
355template <class Node_t>
356int InteracList<Node_t>::class_hash(int* c_){
357 if(!use_symmetries) return coord_hash(c_);
358 int c[3]={abs(c_[0]), abs(c_[1]), abs(c_[2])};
359 if(c[1]>c[0] && c[1]>c[2])
360 {int tmp=c[0]; c[0]=c[1]; c[1]=tmp;}
361 if(c[0]>c[2])
362 {int tmp=c[0]; c[0]=c[2]; c[2]=tmp;}
363 if(c[0]>c[1])
364 {int tmp=c[0]; c[0]=c[1]; c[1]=tmp;}
365 assert(c[0]<=c[1] && c[1]<=c[2]);
366 return coord_hash(&c[0]);
367}

369/**
370 * \brief Set relative coordinates of the interacting node in
371 * rel_coord[Type][idx][1:3].
372 */
373template <class Node_t>
374void InteracList<Node_t>::InitList(int max_r, int min_r, int step, Mat_Type t){
375 size_t count= sctl::pow<unsigned int>((max_r*2)/step+1,dim)
376 -(min_r>0?sctl::pow<unsigned int>((min_r*2)/step-1,dim):0);
377 Matrix<int>& M=rel_coord[t];
378 M.Resize(count,dim);
379 hash_lut[t].assign(PVFMM_MAX_COORD_HASH, -1);
380
381 std::vector<int> class_size_hash(PVFMM_MAX_COORD_HASH, 0);
382 std::vector<int> class_disp_hash(PVFMM_MAX_COORD_HASH, 0);
383 for(int k=-max_r;k<=max_r;k+=step)
384 for(int j=-max_r;j<=max_r;j+=step)
385 for(int i=-max_r;i<=max_r;i+=step)
386 if(abs(i)>=min_r || abs(j)>=min_r || abs(k) >= min_r){
387 int c[3]={i,j,k};
388 class_size_hash[class_hash(c)]++;
389 }
390 omp_par::scan(&class_size_hash[0], &class_disp_hash[0], PVFMM_MAX_COORD_HASH);
391
392 size_t count_=0;
393 for(int k=-max_r;k<=max_r;k+=step)
394 for(int j=-max_r;j<=max_r;j+=step)
395 for(int i=-max_r;i<=max_r;i+=step)
396 if(abs(i)>=min_r || abs(j)>=min_r || abs(k) >= min_r){
397 int c[3]={i,j,k};
398 int& idx=class_disp_hash[class_hash(c)];
399 for(size_t l=0;l<dim;l++) M[idx][l]=c[l];
400 hash_lut[t][coord_hash(c)]=idx;
401 count_++;
402 idx++;
403 }
404 assert(count_==count);
405
406 interac_class[t].resize(count);
407 perm_list[t].resize(count);
408 if(!use_symmetries){ // Set interac_class=self
409 for(size_t j=0;j<count;j++){
410 int c_hash = coord_hash(&M[j][0]);
411 interac_class[t][j]=hash_lut[t][c_hash];
412 }
413 }
414 else for(size_t j=0;j<count;j++){
415 if(M[j][0]<0) perm_list[t][j].push_back(ReflecX);
416 if(M[j][1]<0) perm_list[t][j].push_back(ReflecY);
417 if(M[j][2]<0) perm_list[t][j].push_back(ReflecZ);
418 int coord[3];
419 coord[0]=abs(M[j][0]);
420 coord[1]=abs(M[j][1]);
421 coord[2]=abs(M[j][2]);
422
423 if(coord[1]>coord[0] && coord[1]>coord[2]){
424 perm_list[t][j].push_back(SwapXY);
425 int tmp=coord[0]; coord[0]=coord[1]; coord[1]=tmp;
426 }
427 if(coord[0]>coord[2]){
428 perm_list[t][j].push_back(SwapXZ);
429 int tmp=coord[0]; coord[0]=coord[2]; coord[2]=tmp;
430 }
431 if(coord[0]>coord[1]){
432 perm_list[t][j].push_back(SwapXY);
433 int tmp=coord[0]; coord[0]=coord[1]; coord[1]=tmp;
434 }
435 assert(coord[0]<=coord[1] && coord[1]<=coord[2]);
436 int c_hash = coord_hash(&coord[0]);
437 interac_class[t][j]=hash_lut[t][c_hash];
438 }
439}

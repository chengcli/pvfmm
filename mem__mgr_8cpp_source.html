79void* MemoryManager::malloc(const size_t n_elem, const size_t type_size) const{
80 if(!n_elem) return NULL;
81 static constexpr uintptr_t alignment=PVFMM_MEM_ALIGN-1;
82 static constexpr uintptr_t header_size=(uintptr_t)(sizeof(MemHead)+alignment) & ~(uintptr_t)alignment;
83
84 size_t size=n_elem*type_size+header_size;
85 size=(uintptr_t)(size+alignment) & ~(uintptr_t)alignment;
86 char* base=NULL;
87
88 size_t n_indx;
89 #pragma omp critical(PVFMM_MEM_MGR_CRIT)
90 {
91 //mutex_lock.lock();
92 std::multimap<size_t, size_t>::iterator it=free_map.lower_bound(size);
93 n_indx=(it!=free_map.end()?it->second:0);
94 if(n_indx){ // Allocate from buff
95 size_t n_free_indx=(it->first>size?new_node():0);
96 MemNode& n=node_buff[n_indx-1];
97 assert(n.size==it->first);
98 assert(n.it==it);
99 assert(n.free);
100
101 if(n_free_indx){ // Create a node for the remaining free part.
102 MemNode& n_free=node_buff[n_free_indx-1];
103 n_free=n;
104 n_free.size-=size;
105 n_free.mem_ptr=(char*)n_free.mem_ptr+size;
106 { // Insert n_free to the link list
107 n_free.prev=n_indx;
108 if(n_free.next){
109 size_t n_next_indx=n_free.next;
110 MemNode& n_next=node_buff[n_next_indx-1];
111 n_next.prev=n_free_indx;
112 }
113 n.next=n_free_indx;
114 }
115 assert(n_free.free); // Insert n_free to free map
116 n_free.it=free_map.insert(std::make_pair(n_free.size,n_free_indx));
117 n.size=size; // Update n
118 }
119
120 n.free=false;
121 free_map.erase(it);
122 base = n.mem_ptr;
123 }
124 //mutex_lock.unlock();
125 }
126 if(!base){ // Use system malloc
127 size+=2+alignment;
128 char* p = (char*)DeviceWrapper::host_malloc(size);
129 base = (char*)((uintptr_t)(p+2+alignment) & ~(uintptr_t)alignment);
130 const uint16_t base_offset = (uint16_t)(base-p);
131 *(uint16_t*)(p+(base_offset-sizeof(uint16_t))) = base_offset;
132 }
133
134 { // Check out-of-bounds write
135 #ifndef PVFMM_NDEBUG
136 if(n_indx){
137 #pragma omp parallel for
138 for(size_t i=0;i<size;i++) assert(base[i]==init_mem_val);
139 }
140 #endif
141 }
142
143 MemHead* mem_head=(MemHead*)base;
144 { // Set mem_head
145 mem_head->n_indx=n_indx;
146 mem_head->n_elem=n_elem;
147 mem_head->type_size=type_size;
148 }
149 { // Set header check_sum
150 #ifndef PVFMM_NDEBUG
151 size_t check_sum=0;
152 mem_head->check_sum=0;
153 for(size_t i=0;i<header_size;i++){
154 check_sum+=base[i];
155 }
156 check_sum=check_sum & ((1UL << sizeof(mem_head->check_sum))-1);
157 mem_head->check_sum=check_sum;
158 #endif
159 }
160 return (void*)(base+header_size);
161}

163void MemoryManager::free(void* p) const{
164 if(!p) return;
165 static constexpr uintptr_t alignment=PVFMM_MEM_ALIGN-1;
166 static constexpr uintptr_t header_size=(uintptr_t)(sizeof(MemHead)+alignment) & ~(uintptr_t)alignment;
167
168 char* base=(char*)((char*)p-header_size);
169 MemHead* mem_head=(MemHead*)base;
170
171 if(base<&buff[0] || base>=&buff[buff_size]){ // Use system free
172 char* p_=(char*)((uintptr_t)base-((uint16_t*)base)[-1]);
173 return DeviceWrapper::host_free(p_);
174 }
175
176 size_t n_indx=mem_head->n_indx;
177 assert(n_indx>0 && n_indx<=node_buff.size());
178 { // Verify header check_sum; set array to init_mem_val
179 #ifndef PVFMM_NDEBUG
180 { // Verify header check_sum
181 size_t check_sum=0;
182 for(size_t i=0;i<header_size;i++){
183 check_sum+=base[i];
184 }
185 check_sum-=mem_head->check_sum;
186 check_sum=check_sum & ((1UL << sizeof(mem_head->check_sum))-1);
187 assert(check_sum==mem_head->check_sum);
188 }
189 size_t size=mem_head->n_elem*mem_head->type_size;
190 #pragma omp parallel for
191 for(size_t i=0;i<size;i++) ((char*)p)[i]=init_mem_val;
192 for(size_t i=0;i<sizeof(MemHead);i++) base[i]=init_mem_val;
193 #endif
194 }
195
196 #pragma omp critical(PVFMM_MEM_MGR_CRIT)
197 {
198 //mutex_lock.lock();
199 MemNode& n=node_buff[n_indx-1];
200 assert(!n.free && n.size>0 && n.mem_ptr==base);
201 if(n.prev!=0 && node_buff[n.prev-1].free){
202 size_t n_prev_indx=n.prev;
203 MemNode& n_prev=node_buff[n_prev_indx-1];
204 n.size+=n_prev.size;
205 n.mem_ptr=n_prev.mem_ptr;
206 n.prev=n_prev.prev;
207 free_map.erase(n_prev.it);
208 delete_node(n_prev_indx);
209
210 if(n.prev){
211 node_buff[n.prev-1].next=n_indx;
212 }
213 }
214 if(n.next!=0 && node_buff[n.next-1].free){
215 size_t n_next_indx=n.next;
216 MemNode& n_next=node_buff[n_next_indx-1];
217 n.size+=n_next.size;
218 n.next=n_next.next;
219 free_map.erase(n_next.it);
220 delete_node(n_next_indx);
221
222 if(n.next){
223 node_buff[n.next-1].prev=n_indx;
224 }
225 }
226 n.free=true; // Insert n to free_map
227 n.it=free_map.insert(std::make_pair(n.size,n_indx));
228 //mutex_lock.unlock();
229 }

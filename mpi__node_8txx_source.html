22

79template <class T>
80bool MPI_Node<T>::SubdivCond(){
81 int n=(1UL<<this->Dim());
82 // Do not subdivide beyond max_depth
83 if(this->Depth()>=this->max_depth-1) return false;
84 if(!this->IsLeaf()){ // If has non-leaf children, then return true.
85 for(int i=0;i<n;i++){
86 MPI_Node<Real_t>* ch=static_cast<MPI_Node<Real_t>*>(this->Child(i));
87 assert(ch); //ch==NULL should never happen
88 if(!ch->IsLeaf() || ch->IsGhost()) return true;
89 }
90 }
91 else{ // Do not refine ghost leaf nodes.
92 if(this->IsGhost()) return false;
93 }
94
95 if(!this->IsLeaf()){
96 size_t pt_vec_size=0;
97 for(int i=0;i<n;i++){
98 MPI_Node<Real_t>* ch=static_cast<MPI_Node<Real_t>*>(this->Child(i));
99 pt_vec_size+=ch->pt_coord.Dim();
100 }
101 return pt_vec_size/Dim()>max_pts;
102 }else{
103 return pt_coord.Dim()/Dim()>max_pts;
104 }
105}

107template <class T>
108void MPI_Node<T>::Subdivide(){
109 if(!this->IsLeaf()) return;
110 TreeNode::Subdivide();
111 int nchld=(1UL<<this->Dim());
112
113 if(!IsGhost()){ // Partition point coordinates and values.
114 std::vector<Vector<Real_t>*> pt_coord;
115 std::vector<Vector<Real_t>*> pt_value;
116 std::vector<Vector<size_t>*> pt_scatter;
117 this->NodeDataVec(pt_coord, pt_value, pt_scatter);
118
119 std::vector<std::vector<Vector<Real_t>*> > chld_pt_coord(nchld);
120 std::vector<std::vector<Vector<Real_t>*> > chld_pt_value(nchld);
121 std::vector<std::vector<Vector<size_t>*> > chld_pt_scatter(nchld);
122 for(int i=0;i<nchld;i++){
123 static_cast<MPI_Node<Real_t>*>((MPI_Node<T>*)this->Child(i))
124 ->NodeDataVec(chld_pt_coord[i], chld_pt_value[i], chld_pt_scatter[i]);
125 }
126
127 Real_t* c=this->Coord();
128 Real_t s=sctl::pow<Real_t>(0.5,this->Depth()+1);
129 for(size_t j=0;j<pt_coord.size();j++){
130 if(!pt_coord[j] || !pt_coord[j]->Dim()) continue;
131 Vector<Real_t>& coord=*pt_coord[j];
132 size_t npts=coord.Dim()/this->dim;
133
134 Vector<size_t> cdata(nchld+1);
135 for(int i=0;i<nchld+1;i++){
136 long long pt1=-1, pt2=npts;
137 while(pt2-pt1>1){ // binary search
138 long long pt3=(pt1+pt2)/2;
139 assert(pt3<(long long)npts);
140 if(pt3<0) pt3=0;
141 int ch_id=(coord[pt3*3+0]>=c[0]+s)*1+
142 (coord[pt3*3+1]>=c[1]+s)*2+
143 (coord[pt3*3+2]>=c[2]+s)*4;
144 if(ch_id< i) pt1=pt3;
145 if(ch_id>=i) pt2=pt3;
146 }
147 cdata[i]=pt2;
148 }
149
150 if(pt_coord[j]){
151 Vector<Real_t>& vec=*pt_coord[j];
152 size_t dof=vec.Dim()/npts;
153 if(dof>0) for(int i=0;i<nchld;i++){
154 Vector<Real_t>& chld_vec=*chld_pt_coord[i][j];
155 chld_vec.ReInit((cdata[i+1]-cdata[i])*dof, &vec[0]+cdata[i]*dof);
156 }
157 vec.ReInit(0);
158 }
159 if(pt_value[j]){
160 Vector<Real_t>& vec=*pt_value[j];
161 size_t dof=vec.Dim()/npts;
162 if(dof>0) for(int i=0;i<nchld;i++){
163 Vector<Real_t>& chld_vec=*chld_pt_value[i][j];
164 chld_vec.ReInit((cdata[i+1]-cdata[i])*dof, &vec[0]+cdata[i]*dof);
165 }
166 vec.ReInit(0);
167 }
168 if(pt_scatter[j]){
169 Vector<size_t>& vec=*pt_scatter[j];
170 size_t dof=vec.Dim()/npts;
171 if(dof>0) for(int i=0;i<nchld;i++){
172 Vector<size_t>& chld_vec=*chld_pt_scatter[i][j];
173 chld_vec.ReInit((cdata[i+1]-cdata[i])*dof, &vec[0]+cdata[i]*dof);
174 }
175 vec.ReInit(0);
176 }
177 }
178 }else{
179 std::vector<Vector<Real_t>*> pt_coord;
180 std::vector<Vector<Real_t>*> pt_value;
181 std::vector<Vector<size_t>*> pt_scatter;
182 this->NodeDataVec(pt_coord, pt_value, pt_scatter);
183 for(size_t j=0;j<pt_coord.size();j++){
184 if(pt_coord[j]) pt_coord[j]->ReInit(0);
185 if(pt_value[j]) pt_value[j]->ReInit(0);
186 if(pt_scatter[j]) pt_scatter[j]->ReInit(0);
187 }
188 }

211 ->NodeDataVec(chld_pt_coord[i], chld_pt_value[i], chld_pt_scatter[i]);
212 }
213
214 for(size_t j=0;j<pt_coord.size();j++){
215 if(!pt_coord[j]) continue;
216 if(pt_coord[j]){

275template <class T>
276PackedData MPI_Node<T>::Pack(bool ghost, void* buff_ptr, size_t offset){
277 std::vector<Vector<Real_t>*> pt_coord;
278 std::vector<Vector<Real_t>*> pt_value;
279 std::vector<Vector<size_t>*> pt_scatter;
280 this->NodeDataVec(pt_coord, pt_value, pt_scatter);
281
282 PackedData p0;
283 // Determine data length.
284 p0.length =sizeof(size_t)+sizeof(int)+sizeof(long long)+sizeof(MortonId);
285 for(size_t j=0;j<pt_coord.size();j++){
286 p0.length+=3*sizeof(size_t);
287 if(pt_coord [j]) p0.length+=(pt_coord [j]->Dim())*sizeof(Real_t);
288 if(pt_value [j]) p0.length+=(pt_value [j]->Dim())*sizeof(Real_t);
289 if(pt_scatter[j]) p0.length+=(pt_scatter[j]->Dim())*sizeof(size_t);
290 }
291
292 // Allocate memory.
293 p0.data=(char*)buff_ptr;
294 if(!p0.data){
295 this->packed_data.Resize(p0.length+offset);
296 p0.data=&this->packed_data[0];
297 }
298
299 char* data_ptr=(char*)p0.data;
300 data_ptr+=offset;
301
302 // Header
303 ((size_t*)data_ptr)[0]=p0.length;
304 data_ptr+=sizeof(size_t);
305
306 ((int*)data_ptr)[0]=this->Depth();
307 data_ptr+=sizeof(int);
308
309 ((long long*)data_ptr)[0]=this->NodeCost();
310 data_ptr+=sizeof(long long);
311
312 ((MortonId*)data_ptr)[0]=this->GetMortonId();
313 data_ptr+=sizeof(MortonId);
314
315 // Copy Vector data.
316 for(size_t j=0;j<pt_coord.size();j++){
317 if(pt_coord[j]){
318 Vector<Real_t>& vec=*pt_coord[j];
319 ((size_t*)data_ptr)[0]=vec.Dim(); data_ptr+=sizeof(size_t);
320 if(vec.Dim()>0 && data_ptr!=(char*)&vec[0])
321 mem::copy<Real_t>((Real_t*)data_ptr, &vec[0], vec.Dim());
322 data_ptr+=vec.Dim()*sizeof(Real_t);
323 }else{
324 ((size_t*)data_ptr)[0]=0; data_ptr+=sizeof(size_t);
325 }
326 if(pt_value[j]){
327 Vector<Real_t>& vec=*pt_value[j];
328 ((size_t*)data_ptr)[0]=vec.Dim(); data_ptr+=sizeof(size_t);
329 if(vec.Dim()>0 && data_ptr!=(char*)&vec[0])
330 mem::copy<Real_t>((Real_t*)data_ptr, &vec[0], vec.Dim());
331 data_ptr+=vec.Dim()*sizeof(Real_t);
332 }else{
333 ((size_t*)data_ptr)[0]=0; data_ptr+=sizeof(size_t);
334 }
335 if(pt_scatter[j] && !ghost){
336 Vector<size_t>& vec=*pt_scatter[j];
337 ((size_t*)data_ptr)[0]=vec.Dim(); data_ptr+=sizeof(size_t);
338 if(vec.Dim()>0 && data_ptr!=(char*)&vec[0])
339 mem::copy<size_t>((size_t*)data_ptr, &vec[0], vec.Dim());
340 data_ptr+=vec.Dim()*sizeof(size_t);
341 }else{
342 ((size_t*)data_ptr)[0]=0; data_ptr+=sizeof(size_t);
343 }
344 }
345
346 return p0;
347}

430template <class VTUData_t>
431void AppendVTUData(VTUData_t& vtu_data, VTUData_t& new_data){
432 typedef typename VTUData_t::VTKReal_t VTKReal_t;
433
434 { // Append new_data to vtk_data
435 std::vector<VTKReal_t>& new_coord=new_data.coord;
436 std::vector<std::string>& new_name =new_data.name;
437 std::vector<std::vector<VTKReal_t> >& new_value=new_data.value;
438
439 std::vector<int32_t>& new_connect=new_data.connect;
440 std::vector<int32_t>& new_offset =new_data.offset;
441 std::vector<uint8_t>& new_types =new_data.types;
442
443
444
445 std::vector<VTKReal_t>& vtu_coord=vtu_data.coord;
446 std::vector<std::string>& vtu_name =vtu_data.name;
447 std::vector<std::vector<VTKReal_t> >& vtu_value=vtu_data.value;
448
449 std::vector<int32_t>& vtu_connect=vtu_data.connect;
450 std::vector<int32_t>& vtu_offset =vtu_data.offset;
451 std::vector<uint8_t>& vtu_types =vtu_data.types;
452
453
454
455 size_t old_pts=vtu_coord.size()/PVFMM_COORD_DIM;
456 size_t new_pts=new_coord.size()/PVFMM_COORD_DIM;
457
458 // New points
459 for(size_t i=0;i<new_coord.size();i++){
460 vtu_coord.push_back(new_coord[i]);
461 }
462
463 // Resize old DataArrays
464 for(size_t i=0;i<vtu_value.size();i++){
465 size_t curr_size=vtu_value[i].size();
466 size_t new_size=(curr_size*(old_pts+new_pts))/old_pts;
467 vtu_value[i].resize(new_size,0);
468 }
469
470 // Add new DataArrays
471 for(size_t i=0;i<new_name.size();i++){
472 vtu_name.push_back(new_name[i]);
473 vtu_value.push_back(std::vector<VTKReal_t>());
474 std::vector<VTKReal_t>& value=vtu_value.back();
475 if(new_pts) value.resize((new_value[i].size()*old_pts)/new_pts);
476 for(size_t j=0;j<new_value[i].size();j++){
477 value.push_back(new_value[i][j]);
478 }
479 }
480
481 size_t connect_update=old_pts;
482 size_t offset_update=vtu_connect.size();
483
484 for(size_t i=0;i<new_connect.size();i++){
485 vtu_connect.push_back(connect_update+new_connect[i]);
486 }
487 for(size_t i=0;i<new_offset.size();i++){
488 vtu_offset.push_back(offset_update+new_offset[i]);
489 }
490 for(size_t i=0;i<new_types.size();i++){
491 vtu_types.push_back(new_types[i]);
492 }
493 }

496template <class T>
497template <class VTUData_t, class Node_t>
498void MPI_Node<T>::VTU_Data(VTUData_t& vtu_data, std::vector<Node_t*>& nodes, int lod){
499 typedef typename VTUData_t::VTKReal_t VTKReal_t;
500
501 VTUData_t new_data;
502 { // Set new data
503 new_data.value.resize(1);
504 new_data.name.push_back("pt_value");
505 std::vector<VTKReal_t>& coord=new_data.coord;
506 std::vector<VTKReal_t>& value=new_data.value[0];
507
508 std::vector<int32_t>& connect=new_data.connect;
509 std::vector<int32_t>& offset =new_data.offset;
510 std::vector<uint8_t>& types =new_data.types;
511
512 size_t point_cnt=0;
513 size_t connect_cnt=0;
514 for(size_t nid=0;nid<nodes.size();nid++){
515 Node_t* n=nodes[nid];
516 if(n->IsGhost() || !n->IsLeaf()) continue;
517
518 for(size_t i=0;i<n->pt_coord.Dim();i+=3){
519 coord.push_back(n->pt_coord[i+0]);
520 coord.push_back(n->pt_coord[i+1]);
521 coord.push_back(n->pt_coord[i+2]);
522 connect_cnt++;
523 connect.push_back(point_cnt);
524 offset.push_back(connect_cnt);
525 types.push_back(1);
526 point_cnt++;
527 }
528 for(size_t i=0;i<n->pt_value.Dim();i++){
529 value.push_back(n->pt_value[i]);
530 }
531 }
532 size_t value_dof=(value.size()?value.size()/point_cnt:0);
533 assert(value_dof*point_cnt==value.size());
534 for(size_t nid=0;nid<nodes.size();nid++){
535 Node_t* n=nodes[nid];
536 if(n->IsGhost() || !n->IsLeaf()) continue;
537
538 Real_t* c=n->Coord();
539 Real_t s=sctl::pow<Real_t>(0.5,n->Depth());
540 for(int i=0;i<8;i++){
541 coord.push_back(c[0]+(i&1?1:0)*s);
542 coord.push_back(c[1]+(i&2?1:0)*s);
543 coord.push_back(c[2]+(i&4?1:0)*s);
544 for(int j=0;j<value_dof;j++) value.push_back(0.0);
545 connect.push_back(point_cnt+i);
546 connect_cnt++;
547 }
548 offset.push_back(connect_cnt);
549 types.push_back(11);
550 point_cnt+=8;
551 }
552 }
553 AppendVTUData(vtu_data, new_data);
554}

9template <class T,class StrictWeakOrdering>
10void omp_par::merge(T A_,T A_last,T B_,T B_last,T C_,int p,StrictWeakOrdering comp){
11 typedef typename std::iterator_traits<T>::difference_type _DiffType;
12 typedef typename std::iterator_traits<T>::value_type _ValType;
13
14 _DiffType N1=A_last-A_;
15 _DiffType N2=B_last-B_;
16 if(N1==0 && N2==0) return;
17 if(N1==0 || N2==0){
18 _ValType* A=(N1==0? &B_[0]: &A_[0]);
19 _DiffType N=(N1==0? N2 : N1 );
20 #pragma omp parallel for
21 for(int i=0;i<p;i++){
22 _DiffType indx1=( i *N)/p;
23 _DiffType indx2=((i+1)*N)/p;
24 memcpy(&C_[indx1], &A[indx1], (indx2-indx1)*sizeof(_ValType));
25 }
26 return;
27 }
28
29 //Split both arrays ( A and B ) into n equal parts.
30 //Find the position of each split in the final merged array.
31 int n=10;
32 _ValType* split=mem::aligned_new<_ValType>(p*n*2);
33 _DiffType* split_size=mem::aligned_new<_DiffType>(p*n*2);
34 #pragma omp parallel for
35 for(int i=0;i<p;i++){
36 for(int j=0;j<n;j++){
37 int indx=i*n+j;
38 _DiffType indx1=(indx*N1)/(p*n);
39 split [indx]=A_[indx1];
40 split_size[indx]=indx1+(std::lower_bound(B_,B_last,split[indx],comp)-B_);
41
42 indx1=(indx*N2)/(p*n);
43 indx+=p*n;
44 split [indx]=B_[indx1];
45 split_size[indx]=indx1+(std::lower_bound(A_,A_last,split[indx],comp)-A_);
46 }
47 }
48
49 //Find the closest split position for each thread that will
50 //divide the final array equally between the threads.
51 _DiffType* split_indx_A=mem::aligned_new<_DiffType>(p+1);
52 _DiffType* split_indx_B=mem::aligned_new<_DiffType>(p+1);
53 split_indx_A[0]=0;
54 split_indx_B[0]=0;
55 split_indx_A[p]=N1;
56 split_indx_B[p]=N2;
57 #pragma omp parallel for
58 for(int i=1;i<p;i++){
59 _DiffType req_size=(i*(N1+N2))/p;
60
61 int j=std::lower_bound(&split_size[0],&split_size[p*n],req_size,std::less<_DiffType>())-&split_size[0];
62 if(j>=p*n)
63 j=p*n-1;
64 _ValType split1 =split [j];
65 _DiffType split_size1=split_size[j];
66
67 j=(std::lower_bound(&split_size[p*n],&split_size[p*n*2],req_size,std::less<_DiffType>())-&split_size[p*n])+p*n;
68 if(j>=2*p*n)
69 j=2*p*n-1;
70 if(abs(split_size[j]-req_size)<abs(split_size1-req_size)){
71 split1 =split [j];
72 split_size1=split_size[j];
73 }
74
75 split_indx_A[i]=std::lower_bound(A_,A_last,split1,comp)-A_;
76 split_indx_B[i]=std::lower_bound(B_,B_last,split1,comp)-B_;
77 }
78 mem::aligned_delete(split);
79 mem::aligned_delete(split_size);
80
81 //Merge for each thread independently.
82 #pragma omp parallel for
83 for(int i=0;i<p;i++){
84 T C=C_+split_indx_A[i]+split_indx_B[i];
85 std::merge(A_+split_indx_A[i],A_+split_indx_A[i+1],B_+split_indx_B[i],B_+split_indx_B[i+1],C,comp);
86 }
87 mem::aligned_delete(split_indx_A);
88 mem::aligned_delete(split_indx_B);
89}

91template <class T,class StrictWeakOrdering>
92void omp_par::merge_sort(T A,T A_last,StrictWeakOrdering comp){
93 typedef typename std::iterator_traits<T>::difference_type _DiffType;
94 typedef typename std::iterator_traits<T>::value_type _ValType;
95
96 int p=omp_get_max_threads();
97 _DiffType N=A_last-A;
98 if(N<2*p){
99 std::sort(A,A_last,comp);
100 return;
101 }
102
103 //Split the array A into p equal parts.
104 _DiffType* split=mem::aligned_new<_DiffType>(p+1);
105 split[p]=N;
106 #pragma omp parallel for
107 for(int id=0;id<p;id++){
108 split[id]=(id*N)/p;
109 }
110
111 //Sort each part independently.
112 #pragma omp parallel for
113 for(int id=0;id<p;id++){
114 std::sort(A+split[id],A+split[id+1],comp);
115 }
116
117 //Merge two parts at a time.
118 _ValType* B=mem::aligned_new<_ValType>(N);
119 _ValType* A_=&A[0];
120 _ValType* B_=&B[0];
121 for(int j=1;j<p;j=j*2){
122 for(int i=0;i<p;i=i+2*j){
123 if(i+j<p){
124 omp_par::merge(A_+split[i],A_+split[i+j],A_+split[i+j],A_+split[(i+2*j<=p?i+2*j:p)],B_+split[i],p,comp);
125 }else{
126 #pragma omp parallel for
127 for(int k=split[i];k<split[p];k++)
128 B_[k]=A_[k];
129 }
130 }
131 _ValType* tmp_swap=A_;
132 A_=B_;
133 B_=tmp_swap;
134 }
135
136 //The final result should be in A.
137 if(A_!=&A[0]){
138 #pragma omp parallel for
139 for(int i=0;i<N;i++)
140 A[i]=A_[i];
141 }
142
143 //Free memory.
144 mem::aligned_delete(split);
145 mem::aligned_delete(B);
146}

154template <class T, class I>
155T omp_par::reduce(T* A, I cnt){
156 T sum=0;
157 #pragma omp parallel for reduction(+:sum)

163template <class T, class I>
164void omp_par::scan(T* A, T* B,I cnt){
165 int p=omp_get_max_threads();
166 if(cnt<(I)100*p){
167 for(I i=1;i<cnt;i++)
168 B[i]=B[i-1]+A[i-1];
169 return;
170 }
171
172 I step_size=cnt/p;
173
174 #pragma omp parallel for
175 for(int i=0; i<p; i++){
176 int start=i*step_size;
177 int end=start+step_size;
178 if(i==p-1) end=cnt;
179 if(i!=0)B[start]=0;
180 for(I j=(I)start+1; j<(I)end; j++)
181 B[j]=B[j-1]+A[j-1];
182 }
183
184 T* sum=mem::aligned_new<T>(p);
185 sum[0]=0;
186 for(int i=1;i<p;i++)
187 sum[i]=sum[i-1]+B[i*step_size-1]+A[i*step_size-1];
188
189 #pragma omp parallel for
190 for(int i=1; i<p; i++){
191 int start=i*step_size;
192 int end=start+step_size;
193 if(i==p-1) end=cnt;
194 T sum_=sum[i];
195 for(I j=(I)start; j<(I)end; j++)
196 B[j]+=sum_;
197 }
198 mem::aligned_delete(sum);
199}

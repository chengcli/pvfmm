21 MPI_Comm_rank(comm, &myrank);
22
23 typename ChebFMM_Node<Real>::NodeData tree_data;
24 tree_data.cheb_deg=cheb_deg;
25 tree_data.data_dof=data_dim;
26 tree_data.input_fn=fn_ptr;
27 tree_data.tol=tol;
28 bool adap=true;

31 tree_data.max_depth=PVFMM_MAX_DEPTH;
32 tree_data.max_pts=max_pts;
33
34 { // Set points for initial tree.
35 std::vector<Real> coord;
36 size_t N=sctl::pow<unsigned int>(8,init_depth);
37 N=(N<(size_t)np?np:N)*max_pts;
38 size_t NN=(size_t)ceil(sctl::pow<Real>(N,1.0/3.0));
39 size_t N_total=NN*NN*NN;
40 size_t start= myrank *N_total/np;
41 size_t end =(myrank+1)*N_total/np;
42 for(size_t i=start;i<end;i++){
43 coord.push_back((((i/ 1 )%NN)+(Real)0.5)/NN);
44 coord.push_back((((i/ NN )%NN)+(Real)0.5)/NN);
45 coord.push_back((((i/(NN*NN))%NN)+(Real)0.5)/NN);
46 }
47 tree_data.pt_coord=coord;
48 }
49
50 // Set target points.
51 tree_data.trg_coord=trg_coord;
52
53 auto* tree=new ChebFMM_Tree<Real>(comm);
54 tree->Initialize(&tree_data);
55 tree->InitFMM_Tree(adap,bndry);

59template <class Real>
60inline ChebFMM_Tree<Real>* ChebFMM_CreateTree(int cheb_deg, const std::vector<Real>& node_coord, const std::vector<Real>& fn_coeff, const std::vector<Real>& trg_coord, MPI_Comm comm, BoundaryType bndry){
61 int np, myrank;
62 MPI_Comm_size(comm, &np);
63 MPI_Comm_rank(comm, &myrank);
64
65 typename ChebFMM_Node<Real>::NodeData tree_data;
66 tree_data.input_fn=ChebFn<Real>();
67 tree_data.tol=0;
68 bool adap=false;
69
70 tree_data.dim=PVFMM_COORD_DIM;
71 tree_data.max_depth=PVFMM_MAX_DEPTH;
72 tree_data.max_pts=1;
73
74 tree_data.cheb_deg=cheb_deg;
75 tree_data.pt_value=fn_coeff;
76 tree_data.pt_coord=node_coord;
77 const long Ncoeff = (cheb_deg+1)*(cheb_deg+2)*(cheb_deg+3)/6;
78 { // Set data_dof
79 long long glb_size[2], loc_size[2] = {(long long)node_coord.size()/PVFMM_COORD_DIM, (long long)fn_coeff.size()/Ncoeff};
80 MPI_Allreduce(&loc_size, &glb_size, 2, par::Mpi_datatype<long long>::value(), par::Mpi_datatype<long long>::sum(), comm);
81 tree_data.data_dof = glb_size[1]/glb_size[0];
82 }
83 assert(node_coord.size() && (node_coord.size()/PVFMM_COORD_DIM)*PVFMM_COORD_DIM == node_coord.size());
84 assert(fn_coeff.size() == node_coord.size()/PVFMM_COORD_DIM * tree_data.data_dof * Ncoeff);
85
86 // Set target points.
87 tree_data.trg_coord=trg_coord;
88
89 auto* tree=new ChebFMM_Tree<Real>(comm);
90 tree->Initialize(&tree_data);
91
92 for (auto& node : tree->GetNodeList()) {
93 if(node->IsLeaf() && !node->IsGhost()){
94 node->ChebData() = node->pt_value;
95 node->pt_value.ReInit(0);
96 } else {
97 node->ChebData().ReInit(0);
98 }
99 }
100
101 tree->InitFMM_Tree(adap,bndry);
102 return tree;
103}

122 }
123 trg_value=trg_value_;
124 trg_scatter=trg_scatter_;
125 }
126 par::ScatterReverse(trg_value,trg_scatter,*tree->Comm(),loc_size);
127 trg_val.assign(&trg_value[0],&trg_value[0]+trg_value.Dim());;

156template <class Real>
157void ChebFMM_Coeff2Nodes(std::vector<Real>& node_val, int ChebDeg, int dof, const std::vector<Real>& coeff){
158 const long M0 = (ChebDeg+1)*(ChebDeg+2)*(ChebDeg+3)/6;
159 const long M1 = (ChebDeg+1)*(ChebDeg+1)*(ChebDeg+1);
160 const long N = coeff.size() / M0 / dof;
161 assert(coeff.size() == (size_t)N*M0*dof);
162
163 if (node_val.size() != (size_t)N*M1*dof) node_val.resize(N*M1*dof);
164
165 std::vector<Real> cheb_nds(ChebDeg+1);
166 for (long i = 0; i < ChebDeg+1; i++) cheb_nds[i] = -(Real)cos(M_PI*(i*2+1)/(ChebDeg+1)/2);
167
168 #pragma omp parallel
169 {
170 Vector<Real> buff(dof*M1);
171
172 int np = omp_get_num_threads();
173 int tid = omp_get_thread_num();
174 long a = N*(tid+0)/np;
175 long b = N*(tid+1)/np;
176 for (long i = a; i < b; i++) {
177 const Vector<Real> coeff_(dof*M0, (Real*)coeff.data() + i * dof*M0, false);
178 cheb_eval(coeff_, ChebDeg, cheb_nds, cheb_nds, cheb_nds, buff);
179
180 const Matrix<Real> buff_(dof,M1, buff.Begin(), false);
181 Matrix<Real> node_val_(M1,dof, (Real*)node_val.data() + i * M1*dof, false);
182 Matrix<Real>::Transpose(node_val_, buff_);
183 }
184 }
185}

189 const long M0 = (ChebDeg+1)*(ChebDeg+2)*(ChebDeg+3)/6;
190 const long M1 = (ChebDeg+1)*(ChebDeg+1)*(ChebDeg+1);
191 const long N = node_val.size() / M1 / dof;
192 assert(node_val.size() == (size_t)N*M1*dof);
193
194 if (coeff.size() != (size_t)N*M0*dof) coeff.resize(N*M0*dof);
195
196 #pragma omp parallel
197 {
198 Vector<Real> buff(dof*M1);
199
200 int np = omp_get_num_threads();
201 int tid = omp_get_thread_num();
202 long a = N*(tid+0)/np;
203 long b = N*(tid+1)/np;
204 for (long i = a; i < b; i++) {
205 const Matrix<Real> node_val_(M1,dof, (Real*)node_val.data() + i * M1*dof, false);
206 Matrix<Real> buff_(dof,M1, buff.Begin(), false);
207 Matrix<Real>::Transpose(buff_, node_val_);
208
209 cheb_approx<Real,Real>(buff.Begin(), ChebDeg, dof, coeff.data()+i*dof*M0);
210 }
211 }

257inline void PtFMM_Evaluate(const PtFMM_Tree<Real>* tree, std::vector<Real>& trg_val, size_t loc_size, const std::vector<Real>* src_val, const std::vector<Real>* surf_val){
258 if(src_val){
259 std::vector<size_t> src_scatter_;
260 const auto& nodes=((PtFMM_Tree<Real>*)tree)->GetNodeList();

278 indx++;

298 for(size_t i=0;i<nodes.size();i++){
299 if(nodes[i]->IsLeaf() && !nodes[i]->IsGhost()){
300 Vector<Real>& surf_value_=nodes[i]->surf_value;
301 for(size_t j=0;j<surf_value_.Dim();j++){
302 surf_value_[j]=surf_value[indx];

304 }

306 }
307 }

309 Vector<Real> trg_value;
310 Vector<size_t> trg_scatter;
311 {
312 std::vector<Real> trg_value_;
313 std::vector<size_t> trg_scatter_;

315 for(size_t i=0;i<nodes.size();i++){
316 if(nodes[i]->IsLeaf() && !nodes[i]->IsGhost()){
317 const auto& trg_value=nodes[i]->trg_value;
318 const auto& trg_scatter=nodes[i]->trg_scatter;
319 for(size_t j=0;j<trg_value.Dim();j++) trg_value_.push_back(trg_value[j]);
320 for(size_t j=0;j<trg_scatter.Dim();j++) trg_scatter_.push_back(trg_scatter[j]);
321 }
322 }
323 trg_value=trg_value_;
324 trg_scatter=trg_scatter_;
325 }
326 par::ScatterReverse(trg_value,trg_scatter,*tree->Comm(),loc_size);
327 trg_val.assign(&trg_value[0],&trg_value[0]+trg_value.Dim());;
328}

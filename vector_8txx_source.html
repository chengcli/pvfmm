126void Vector<T>::ReInit(size_t dim_, T* data_, bool own_data_){
127 if(own_data_ && own_data && dim_<=capacity){
128 if(dim!=dim_) FreeDevice(false);
129 dim=dim_;
130 if(data_) mem::copy<T>(data_ptr,data_,dim);
131 }else{
132 Vector<T> tmp(dim_,data_,own_data_);

138typename Vector<T>::Device& Vector<T>::AllocDevice(bool copy){
139 if(dev.dev_ptr==(uintptr_t)NULL && dim>0) // Allocate data on device.
140 dev.dev_ptr=DeviceWrapper::alloc_device((char*)data_ptr, dim*sizeof(T));
141 if(dev.dev_ptr!=(uintptr_t)NULL && copy) // Copy data to device
142 DeviceWrapper::host2device((char*)data_ptr,(char*)data_ptr,dev.dev_ptr,dim*sizeof(T));
